\section{Online Prediction under Information Partitions}
\label{sec:main-setup}

Let $\Omega$ be a finite case space, $\omega\sim P$, $Y:\Omega\to\calY$ a consequence variable, $\mathsf A$ an action space, and $L:\calY\times\mathsf A\to\R$ a loss with finite expectations. An information structure is a partition $\Pi=\{B_1,\ldots,B_m\}$. We write $\Pi\preceq\Pi'$ when $\Pi'$ refines $\Pi$. A predictor is $\Pi$-measurable when it is constant inside each atom:
\begin{align*}
\Pred(\Pi)
=
\{f:\Omega\to\mathsf A:
\omega,\omega'\in B\in\Pi \Rightarrow f(\omega)=f(\omega')\}.
\end{align*}
The Bayes risk under $\Pi$ is
\begin{align*}
R_L(\Pi)
=
\inf_{f\in\Pred(\Pi)}\E[L(Y,f(\omega))]
=
\sum_{B\in\Pi}P(B)\inf_{a\in\mathsf A}\E[L(Y,a)\mid \omega\in B].
\end{align*}
A repair update changes the payload $a(B)$ assigned to existing atoms; a refinement changes the equivalence relation itself. If a runtime representation is an information map $I:\Omega\to\calZ$, then it induces $\Pi(I)=\{I^{-1}(z):I^{-1}(z)\ne\varnothing\}$, and every predictor of the form $a(I(\omega))$ is $\Pi(I)$-measurable. This is the bridge to neural or memory implementations: the theorem constrains the quotient induced by the runtime state, not the syntax of the optimizer used to update that state. All statements extend to sub-sigma-algebras; the full measure-theoretic version and base-information convention are in Appendix~\ref{sec:preliminaries}.

A binary distinction is a prediction-time observable map $U:\Omega\to\{0,1\}$. We assume admissible distinctions lie in a pre-outcome observable class $\calU$, so $U_t$ does not depend on $Y_t$ after observation. The join refinement is
\begin{align*}
\Pi\vee U
=
\{B\cap U^{-1}(0),B\cap U^{-1}(1):B\in\Pi\}\setminus\{\varnothing\}.
\end{align*}

\begin{definition}[Repair and refinement]
\label{def:main-repair-refinement}
At round $t$, the learner predicts with $\hat f_t\in\Pred(\Pi_t)$, observes $Y_t$, and then chooses one of two update types. A repair update has $\Pi_{t+1}=\Pi_t$ and may change only atom payloads, local memories, sufficient statistics, or downstream parameters. A refinement update has $\Pi_{t+1}=\Pi_t\vee U_t$ for some admissible distinction $U_t\in\calU$. The optional cost model charges $\lambda c(U_t)$ only for refinement.
\end{definition}

\begin{theorem}[Risk decomposition, informal version of Theorem~\ref{thm:error-decomposition}]
\label{thm:main-decomposition}
For any $f_t\in\Pred(\Pi_t)$ and any comparator $\Pi^\star$ with $\Pi_t\preceq\Pi^\star$,
\begin{align*}
R_L(f_t)-R_L(\Pi^\star)
=
\underbrace{R_L(f_t)-R_L(\Pi_t)}_{\Delta_{\rm repair}}
+
\underbrace{R_L(\Pi_t)-R_L(\Pi^\star)}_{\Delta_{\rm refine}}.
\end{align*}
Both terms are nonnegative.
\end{theorem}

\begin{proof}[Proof sketch]
\begin{align*}
R_L(f_t)-R_L(\Pi^\star)
&=
R_L(f_t)-R_L(\Pi_t)
+
R_L(\Pi_t)-R_L(\Pi^\star).
\end{align*}
Since $f_t\in\Pred(\Pi_t)$,
\begin{align*}
R_L(f_t)\ge R_L(\Pi_t).
\end{align*}
Since $\Pi^\star$ refines $\Pi_t$,
\begin{align*}
\Pred(\Pi_t)\subseteq \Pred(\Pi^\star),
\qquad
R_L(\Pi_t)\ge R_L(\Pi^\star).
\end{align*}
\end{proof}

The first term is residual risk at fixed information; the second is the population value of changing information. Thus repair versus refinement is a choice between admissible prediction classes. The theorem below turns this distinction into a sequential lower bound.

\begin{theorem}[Repair-only linear regret, informal version of Theorem~\ref{thm:repair-only-regret}]
\label{thm:main-repair-only}
Let $\Pi$ be fixed and let $\Pi\preceq\Pi^\star$. Consider any adaptive learner such that, for every $t$, $\hat f_t\in\Pred(\Pi)$, possibly chosen using all previous observations. If $\omega_t\stackrel{\rm iid}{\sim}P$, then
\begin{align*}
\E\left[\sum_{t=1}^T L(Y_t,\hat f_t(\omega_t))\right]
\ge
T R_L(\Pi).
\end{align*}
If Bayes actions exist under $\Pi^\star$, then regret against the refined oracle satisfies
\begin{align*}
\E[\Reg_T(\Pi^\star)]
\ge
T\left(R_L(\Pi)-R_L(\Pi^\star)\right).
\end{align*}
If the infimum under $\Pi^\star$ is not attained, then for every $\epsilon>0$ the same statement holds against an $\epsilon$-optimal refined oracle with right side $T(R_L(\Pi)-R_L(\Pi^\star)-\epsilon)$. If refinement cost is charged once as $\lambda C(\Pi^\star)$, subtract $\lambda C(\Pi^\star)$ from the bound.
\end{theorem}

\begin{proof}[Proof sketch]
Condition on $\calH_{t-1}$. Since $\hat f_t\in\Pred(\Pi)$,
\begin{align*}
\E[L(Y_t,\hat f_t(\omega_t))\mid \calH_{t-1}]
\ge R_L(\Pi).
\end{align*}
Therefore
\begin{align*}
\E\sum_{t=1}^T L(Y_t,\hat f_t(\omega_t))
\ge
T R_L(\Pi).
\end{align*}
If the refined oracle attains $R_L(\Pi^\star)$ per round, subtraction gives
\begin{align*}
\E[\Reg_T(\Pi^\star)]
\ge
T(R_L(\Pi)-R_L(\Pi^\star)).
\end{align*}
For unattained infima, replace the oracle by an $\epsilon$-optimal $\Pi^\star$-measurable predictor; a one-time refinement charge subtracts from the comparator advantage.
\end{proof}

Once the quotient is fixed, no optimizer can exploit distinctions outside it. The refined oracle is intentionally generous: it receives $\Pi^\star$ for free except for the optional one-time cost. The result therefore does not claim that a refinement algorithm will find $\Pi^\star$ quickly; it says that refusing to change the quotient pays the information gap every round.

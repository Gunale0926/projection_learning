\section{Minimal Refinement Proofs}
\label{app:refinement-proofs}

The statements in this appendix use standard facts about finite partitions, generated sigma-algebras, and measurability \citep{billingsley1995probability,kallenberg2002foundations}. The contribution is the runtime-learning interpretation: refinement is the minimal information update needed by new evidence.

\begin{theorem}[Minimal finite refinement, formal version of Theorem~\ref{thm:minimal-refinement:informal}]
\label{thm:minimal-refinement:formal}
Let $\Omega$ be finite, let $\calP$ be a partition of $\Omega$, and let $E\subseteq\Omega$. Define
\begin{align*}
\calP_E=\{B\cap E:B\in\calP, B\cap E\ne\varnothing\}\cup\{B\cap E^c:B\in\calP, B\cap E^c\ne\varnothing\}.
\end{align*}
Then:
\begin{itemize}
\item {\bf Part 1.} $\calP_E$ is a partition of $\Omega$.
\item {\bf Part 2.} $\calP_E$ refines $\calP$.
\item {\bf Part 3.} $E$ is $\calP_E$-measurable.
\item {\bf Part 4.} If $\calQ$ refines $\calP$ and $E$ is $\calQ$-measurable, then $\calQ$ refines $\calP_E$.
\end{itemize}
Consequently, $\calP_E$ is the unique minimal refinement of $\calP$ making $E$ measurable.
\end{theorem}
\begin{proof}
For each $B\in\calP$, the nonempty sets among $B\cap E$ and $B\cap E^c$ are disjoint and have union $B$. Since the atoms of $\calP$ are disjoint and cover $\Omega$, the atoms of $\calP_E$ are disjoint and cover $\Omega$. Thus Part 1 holds.

Every atom of $\calP_E$ is contained in some $B\in\calP$, so Part 2 holds. Also
\begin{align*}
E=\bigcup_{B\in\calP}(B\cap E),
\end{align*}
where empty intersections are ignored. Hence $E$ is a union of atoms of $\calP_E$, proving Part 3.

Let $\calQ$ refine $\calP$ and make $E$ measurable. For any atom $Q\in\calQ$, there exists $B\in\calP$ with
\begin{align*}
Q\subseteq B.
\end{align*}
Since $E$ is $\calQ$-measurable, each atom of $\calQ$ is either contained in $E$ or disjoint from $E$. Hence
\begin{align*}
Q\subseteq B\cap E\qquad \text{or}\qquad Q\subseteq B\cap E^c.
\end{align*}
Thus each atom of $\calQ$ is contained in an atom of $\calP_E$, so $\calQ$ refines $\calP_E$. This proves Part 4.

If another partition has the same minimal property, it refines $\calP_E$ and is refined by $\calP_E$, so the two partitions have the same atoms.
\end{proof}

\begin{theorem}[Minimal sigma-algebra refinement, formal version of Theorem~\ref{thm:sigma-refinement:informal}]
\label{thm:sigma-refinement:formal}
Let $(\Omega,\calA)$ be a measurable space, let $\calF\subseteq\calA$ be a sigma-algebra, and let $Z:\Omega\to\calZ$ be a measurable map into a measurable space $(\calZ,\calB)$. The join $\calF\vee\sigma(Z)$ is the smallest sigma-algebra $\calG\subseteq\calA$ such that $\calF\subseteq\calG$ and $Z$ is $\calG$-measurable.
\end{theorem}
\begin{proof}
By definition,
\begin{align*}
\calF\vee\sigma(Z)=\sigma(\calF\cup\sigma(Z)).
\end{align*}
Therefore $\calF\subseteq\calF\vee\sigma(Z)$ and $\sigma(Z)\subseteq\calF\vee\sigma(Z)$, so $Z$ is measurable with respect to $\calF\vee\sigma(Z)$.

Let $\calG$ be any sigma-algebra such that $\calF\subseteq\calG$ and $Z$ is $\calG$-measurable. The latter condition means $\sigma(Z)\subseteq\calG$. Hence
\begin{align*}
\calF\cup\sigma(Z)\subseteq\calG.
\end{align*}
Since $\calG$ is a sigma-algebra,
\begin{align*}
\sigma(\calF\cup\sigma(Z))\subseteq\calG.
\end{align*}
Thus $\calF\vee\sigma(Z)\subseteq\calG$.
\end{proof}

\begin{theorem}[Contradiction requires distinction, formal version of Theorem~\ref{thm:contradiction:informal}]
\label{thm:contradiction:formal}
Let $\calP$ be a partition of $\Omega$ and let $y:\Omega\to A$ be a required output. Suppose there exist $\omega,\omega'\in B$ for some atom $B\in\calP$ such that $y(\omega)\ne y(\omega')$. Then no $\calP$-measurable $f:\Omega\to A$ satisfies both
\begin{align*}
f(\omega)=y(\omega),\qquad f(\omega')=y(\omega').
\end{align*}
\end{theorem}
\begin{proof}
Since $f$ is $\calP$-measurable and $\omega,\omega'\in B$,
\begin{align*}
f(\omega)=f(\omega').
\end{align*}
If both requirements held, then
\begin{align*}
y(\omega)=f(\omega)=f(\omega')=y(\omega'),
\end{align*}
contradicting $y(\omega)\ne y(\omega')$.
\end{proof}

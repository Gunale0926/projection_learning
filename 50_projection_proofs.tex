\section{Minimal Refinement Proofs}
\label{app:refinement-proofs}

The statements in this appendix use standard facts about finite partitions, generated sigma-algebras, and measurability \citep{billingsley1995probability,kallenberg2002foundations}. The contribution is the runtime-learning interpretation: refinement is the minimal information update needed by new evidence.

\begin{theorem}[Minimal finite refinement, formal version of Theorem~\ref{thm:minimal-refinement:informal}]
\label{thm:minimal-refinement:formal}
Let $\Omega$ be finite, let $\calP$ be a partition of $\Omega$, and let $D\subseteq\Omega$. Define
\begin{align*}
\calP_D=\{B\cap D:B\in\calP, B\cap D\ne\varnothing\}\cup\{B\cap D^c:B\in\calP, B\cap D^c\ne\varnothing\}.
\end{align*}
Then:
\begin{itemize}
\item {\bf Part 1.} $\calP_D$ is a partition of $\Omega$.
\item {\bf Part 2.} $\calP_D$ refines $\calP$.
\item {\bf Part 3.} $D$ is $\calP_D$-measurable.
\item {\bf Part 4.} If $\calQ$ refines $\calP$ and $D$ is $\calQ$-measurable, then $\calQ$ refines $\calP_D$.
\end{itemize}
Consequently, $\calP_D$ is the unique minimal refinement of $\calP$ making $D$ measurable.
\end{theorem}
\begin{proof}
For each $B\in\calP$, the nonempty sets among $B\cap D$ and $B\cap D^c$ are disjoint and have union $B$. Since the atoms of $\calP$ are disjoint and cover $\Omega$, the atoms of $\calP_D$ are disjoint and cover $\Omega$. Thus Part 1 holds.

Every atom of $\calP_D$ is contained in some $B\in\calP$, so Part 2 holds. Also
\begin{align*}
D=\bigcup_{B\in\calP}(B\cap D),
\end{align*}
where empty intersections are ignored. Hence $D$ is a union of atoms of $\calP_D$, proving Part 3.

Let $\calQ$ refine $\calP$ and make $D$ measurable. For any atom $Q\in\calQ$, there exists $B\in\calP$ with
\begin{align*}
Q\subseteq B.
\end{align*}
Since $D$ is $\calQ$-measurable, each atom of $\calQ$ is either contained in $D$ or disjoint from $D$. Hence
\begin{align*}
Q\subseteq B\cap D\qquad \text{or}\qquad Q\subseteq B\cap D^c.
\end{align*}
Thus each atom of $\calQ$ is contained in an atom of $\calP_D$, so $\calQ$ refines $\calP_D$. This proves Part 4.

If another partition has the same minimal property, it refines $\calP_D$ and is refined by $\calP_D$, so the two partitions have the same atoms.
\end{proof}

\begin{theorem}[Minimal sigma-algebra refinement, formal version of Theorem~\ref{thm:sigma-refinement:informal}]
\label{thm:sigma-refinement:formal}
Let $(\Omega,\calE)$ be a measurable space, let $\calF\subseteq\calE$ be a sigma-algebra, and let $U:\Omega\to\calU$ be a measurable map into a measurable space $(\calU,\Sigma_{\calU})$. The join $\calF\vee\sigma(U)$ is the smallest sigma-algebra $\calG\subseteq\calE$ such that $\calF\subseteq\calG$ and $U$ is $\calG$-measurable.
\end{theorem}
\begin{proof}
By definition,
\begin{align*}
\calF\vee\sigma(U)=\sigma(\calF\cup\sigma(U)).
\end{align*}
Therefore $\calF\subseteq\calF\vee\sigma(U)$ and $\sigma(U)\subseteq\calF\vee\sigma(U)$, so $U$ is measurable with respect to $\calF\vee\sigma(U)$.

Let $\calG$ be any sigma-algebra such that $\calF\subseteq\calG$ and $U$ is $\calG$-measurable. The latter condition means $\sigma(U)\subseteq\calG$. Hence
\begin{align*}
\calF\cup\sigma(U)\subseteq\calG.
\end{align*}
Since $\calG$ is a sigma-algebra,
\begin{align*}
\sigma(\calF\cup\sigma(U))\subseteq\calG.
\end{align*}
Thus $\calF\vee\sigma(U)\subseteq\calG$.
\end{proof}

\begin{theorem}[Contradiction requires distinction, formal version of Corollary~\ref{thm:contradiction:informal}]
\label{thm:contradiction:formal}
Let $I:\Omega\to\calZ$ be an information map and let $y:\Omega\to\mathsf A$ be a required output. Suppose $I(\omega)=I(\omega')$ but $y(\omega)\ne y(\omega')$. Then no predictor $f=g\circ I$ satisfies both
\begin{align*}
f(\omega)=y(\omega),\qquad f(\omega')=y(\omega').
\end{align*}
\end{theorem}
\begin{proof}
Since $f=g\circ I$ and $I(\omega)=I(\omega')$,
\begin{align*}
f(\omega)=f(\omega').
\end{align*}
If both requirements held, then
\begin{align*}
y(\omega)=f(\omega)=f(\omega')=y(\omega'),
\end{align*}
contradicting $y(\omega)\ne y(\omega')$.
\end{proof}

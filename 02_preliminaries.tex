\section{Online Prediction with Refinable Partitions}
\label{sec:preliminaries}

We start with a finite model. The measurable-space notation used later is the direct sigma-algebra generalization, but the computational object is a partition. A runtime state declares which cases are equivalent; a repair changes predictions on equivalence classes, while a refinement changes the equivalence relation.

\subsection{Partitions, predictors, and risk}

Let $\Omega$ be a finite case space, let $P$ be a distribution on $\Omega$, let $Y:\Omega\to\calY$ be the consequence variable, let $\mathsf A$ be an action space, and let $L:\calY\times\mathsf A\to\R$ be a loss. A finite partition
\begin{align*}
\Pi=\{B_1,\ldots,B_m\}
\end{align*}
induces the equivalence relation $\omega\sim_\Pi\omega'$ iff $\omega,\omega'$ lie in the same atom $B\in\Pi$. A predictor under $\Pi$ is a map $a:\Pi\to\mathsf A$, equivalently a function $f:\Omega\to\mathsf A$ that is constant on every atom:
\begin{align*}
\Pred(\Pi)
=
\{f:\Omega\to\mathsf A:
\omega\sim_\Pi\omega'\Rightarrow f(\omega)=f(\omega')\}.
\end{align*}
The Bayes risk achievable under partition $\Pi$ is
\begin{align}
R_L(\Pi)
=
\inf_{f\in\Pred(\Pi)}
\E[L(Y,f)]
=
\sum_{B\in\Pi}P(B)
\inf_{a\in\mathsf A}\E[L(Y,a)\mid \omega\in B],
\label{eq:partition-bayes-risk}
\end{align}
where atoms with $P(B)=0$ may be assigned arbitrary actions. Thus $R_L(\Pi)$ is the best possible risk after the representation has committed to the equivalence relation $\sim_\Pi$.

\begin{definition}[Refinable online prediction problem]
\label{def:refinable-online-prediction}
A refinable online prediction problem is a tuple
\begin{align*}
\mathfrak P=(\Omega,\calY,\mathsf A,P,L,\calU,c),
\end{align*}
where $\Omega$ is finite, $Y:\Omega\to\calY$ is the consequence variable, $P$ is the sampling distribution, $L:\calY\times\mathsf A\to[0,1]$ is a bounded loss, $\calU\subseteq\{0,1\}^{\Omega}$ is a class of admissible binary distinctions, and $c:\calU\to\R_+$ assigns refinement costs.

At round $t$, the learner maintains $S_t=(\Pi_t,a_t)$ with $a_t:\Pi_t\to\mathsf A$. A case $\omega_t\sim P$ is drawn, the learner observes only the atom $B_t\in\Pi_t$ containing $\omega_t$, predicts $a_t(B_t)$, observes $Y_t=Y(\omega_t)$, and then chooses one of two updates:
\begin{align*}
\mathrm{Repair}:&\qquad \Pi_{t+1}=\Pi_t,\quad a_{t+1}\text{ may change},\\
\mathrm{Refine}:&\qquad \Pi_{t+1}=\Pi_t\vee U_t\text{ for some }U_t\in\calU.
\end{align*}
For a binary distinction $U:\Omega\to\{0,1\}$,
\begin{align}
\Pi\vee U
=
\{B\cap U^{-1}(0),B\cap U^{-1}(1):B\in\Pi\}\setminus\{\varnothing\}.
\label{eq:partition-join}
\end{align}
The cumulative objective is prediction loss plus refinement cost,
\begin{align*}
\calL_T
=
\sum_{t=1}^T L(Y_t,a_t(B_t))
+
\lambda\sum_{t:\mathrm{Refine}}c(U_t).
\end{align*}
\end{definition}

The partition formulation also covers information maps. If a runtime representation is $I_t:\Omega\to\calZ_t$, its induced partition is the set of nonempty fibers $I_t^{-1}(z)$. Repair preserves these fibers. Refinement splits at least one fiber.

\subsection{Repair and refinement gaps}

For a concrete predictor $f\in\Pred(\Pi)$ and a candidate refined partition $\Pi^\star$ that refines $\Pi$, define
\begin{align}
\Delta_{\mathrm{repair}}(f,\Pi)
&=R_L(f)-R_L(\Pi),
\label{eq:repair-gap}\\
\Delta_{\mathrm{refine}}(\Pi,\Pi^\star)
&=R_L(\Pi)-R_L(\Pi^\star).
\label{eq:refinement-gap}
\end{align}
The first gap is the residual error of the current predictor at fixed equivalence relation. The second is the value of changing the equivalence relation.

\begin{theorem}[Runtime error decomposition]
\label{thm:error-decomposition}
Let $f_t\in\Pred(\Pi_t)$ and let $\Pi^\star$ refine $\Pi_t$. Then
\begin{align}
R_L(f_t)-R_L(\Pi^\star)
=
\Delta_{\mathrm{repair}}(f_t,\Pi_t)
+
\Delta_{\mathrm{refine}}(\Pi_t,\Pi^\star).
\label{eq:error-decomposition}
\end{align}
Both terms are nonnegative.
\end{theorem}

\begin{proof}
Add and subtract $R_L(\Pi_t)$. The repair gap is nonnegative because $f_t$ is feasible for the infimum defining $R_L(\Pi_t)$. The refinement gap is nonnegative because every $\Pi_t$-measurable predictor is also measurable with respect to any refinement $\Pi^\star$.
\end{proof}

\subsection{Repair-only lower bound}

The pointwise contradiction inside a cell becomes a sequential lower bound. If the current partition collapses states that a refined oracle separates, no repair-only learner can avoid paying the corresponding Bayes-risk gap every round.

\begin{theorem}[Repair-only linear regret]
\label{thm:repair-only-regret}
Let $\Pi$ be a fixed partition and let $\Pi^\star$ be a refinement. Consider any learner whose round-$t$ prediction $\hat f_t$ is always in $\Pred(\Pi)$, possibly chosen adaptively from all previous observations. Then, for iid $\omega_t\sim P$,
\begin{align*}
\E\left[\sum_{t=1}^T L(Y_t,\hat f_t(\omega_t))\right]
\ge
T R_L(\Pi).
\end{align*}
Consequently, relative to an oracle using $\Pi^\star$,
\begin{align*}
\E[\mathrm{Regret}_T]
\ge
T\left(R_L(\Pi)-R_L(\Pi^\star)\right).
\end{align*}
If $R_L(\Pi)>R_L(\Pi^\star)$, every repair-only learner suffers $\Omega(T)$ expected regret against the refined oracle.
\end{theorem}

\begin{proof}
Condition on the learner's history before round $t$. The chosen predictor $\hat f_t$ is $\Pi$-measurable, so by definition of $R_L(\Pi)$,
\begin{align*}
\E[L(Y_t,\hat f_t(\omega_t))\mid \text{history}]
\ge
R_L(\Pi).
\end{align*}
Taking expectation and summing over $t$ gives the first inequality. The $\Pi^\star$ oracle achieves $R_L(\Pi^\star)$ per round by using the Bayes action on each atom of $\Pi^\star$, so subtracting its cumulative risk gives the regret bound.
\end{proof}

\begin{example}[Two-state separation]
\label{ex:two-state-separation}
Let $\Omega=\{0,1\}$, $P(0)=P(1)=1/2$, and $Y=\omega$. Under the trivial partition $\Pi=\{\{0,1\}\}$, the best log-loss predictor is $q(Y=0)=q(Y=1)=1/2$, so $R_{\log}(\Pi)=\log 2$. The refined partition $\Pi^\star=\{\{0\},\{1\}\}$ has $R_{\log}(\Pi^\star)=0$ in the noiseless case. Therefore any repair-only learner has expected regret $T\log 2$ against the refined oracle. This example is not about neural networks, caches, or trees. It is about equivalence relations: a learner that identifies two states cannot assign them two different consequences.
\end{example}

\begin{corollary}[Contradiction requires distinction]
\label{thm:contradiction:informal}
Let $\Pi$ be a partition and let $y:\Omega\to\mathsf A$ be a required deterministic output. If $\omega,\omega'$ lie in the same atom of $\Pi$ but $y(\omega)\ne y(\omega')$, then no predictor in $\Pred(\Pi)$ satisfies both requirements.
\end{corollary}

\begin{proof}
Every $f\in\Pred(\Pi)$ is constant on the atom containing $\omega$ and $\omega'$, so $f(\omega)=f(\omega')$. Satisfying both requirements would imply $y(\omega)=y(\omega')$, a contradiction.
\end{proof}

\subsection{Sigma-algebra extension}

The finite partition model is the main computational model. The sigma-algebra notation is the measure-theoretic extension used for value identities and language-model interpretations. On a probability space $(\Omega,\calE,\mathbb P)$, an information structure is a sub-sigma-algebra $\calF\subseteq\calE$, and
\begin{align*}
\Pred(\calF)
=
\{a:\Omega\to\mathsf A:a\text{ is }\calF\text{-measurable}\},
\qquad
R_L(\calF)=\inf_{a\in\Pred(\calF)}\E[L(Y,a)].
\end{align*}
For a finite partition $\Pi$, this recovers the previous definitions by taking $\calF=\sigma(\Pi)$.

When the infimum is attained, write $a_\calF^\star$ for a Bayes action under $\calF$. A richer information structure is useful only when its optimal action cannot already be expressed using the old information.

\begin{theorem}[Loss-relevant refinement]
\label{thm:loss-relevant-refinement:informal}
Let $\calF\subseteq\calG\subseteq\calE$. Assume Bayes actions exist under $\calF$ and $\calG$, and that the $\calG$-Bayes action is almost surely unique. Then
\begin{align*}
R_L(\calF)-R_L(\calG)=0
\end{align*}
if and only if $a_\calG^\star$ has an $\calF$-measurable version.
\end{theorem}

\begin{proof}
If $a_\calG^\star$ is $\calF$-measurable, then it is feasible for $R_L(\calF)$, so
\begin{align*}
R_L(\calF)\le \E[L(Y,a_\calG^\star)]=R_L(\calG).
\end{align*}
Monotonicity under $\calF\subseteq\calG$ gives the reverse inequality. Conversely, if the risks are equal and $a_\calF^\star$ is an $\calF$-Bayes action, then $a_\calF^\star$ is also feasible under $\calG$ and achieves $R_L(\calG)$. By almost sure uniqueness under $\calG$, $a_\calF^\star=a_\calG^\star$ almost surely. Hence $a_\calG^\star$ has an $\calF$-measurable version.
\end{proof}

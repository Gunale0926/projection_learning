\section{Value of Distinction Proofs}
\label{app:value-proofs}

The proofs below use standard tools from statistical decision theory, information theory, proper scoring rules, and conditional expectation: Bayes risk under constrained information, log-loss optimality of the conditional law, and the $L^2$ projection identity for conditional expectation \citep{blackwell1953equivalent,berger1985statistical,cover2006elements,gneiting2007strictly,billingsley1995probability,kallenberg2002foundations}.

\begin{theorem}[Bayes-risk monotonicity, formal version of Theorem~\ref{thm:risk-monotonicity:informal}]
\label{thm:risk-monotonicity:formal}
Let $Y$ be a target random variable, let $\mathsf A$ be an action space, and let $L$ be an integrable loss. If $\calF\subseteq\calG$, then
\begin{align*}
R_L(\calG)\le R_L(\calF).
\end{align*}
\end{theorem}
\begin{proof}
Define
\begin{align*}
\calD_\calF=\{a:\Omega\to\mathsf A:a\text{ is }\calF\text{-measurable}\},\qquad \calD_\calG=\{a:\Omega\to\mathsf A:a\text{ is }\calG\text{-measurable}\}.
\end{align*}
Since $\calF\subseteq\calG$,
\begin{align*}
\calD_\calF\subseteq\calD_\calG.
\end{align*}
Therefore
\begin{align*}
R_L(\calG)&=\inf_{a\in\calD_\calG}\E[L(Y,a)]\\
&\le\inf_{a\in\calD_\calF}\E[L(Y,a)]\\
&=R_L(\calF).
\end{align*}
\end{proof}

\begin{theorem}[Log-loss value, formal version of Theorem~\ref{thm:log-value:informal}]
\label{thm:log-value:formal}
Let $Y$ be finite and let $\calF\subseteq\calG$. Under log loss, an admissible action is a measurable conditional distribution $q(\cdot)$ on the range of $Y$, and the loss is $-\log q(Y)$. Then
\begin{align*}
R_{\log}(\calF)-R_{\log}(\calG)=I(Y;\calG\mid\calF).
\end{align*}
\end{theorem}
\begin{proof}
For any $\calF$-measurable conditional distribution $q$, write $p_\calF(y)=\mathbb P(Y=y\mid\calF)$. Then
\begin{align*}
\E[-\log q(Y)\mid\calF]&=\sum_y p_\calF(y)(-\log q(y))\\
&=\sum_y p_\calF(y)(-\log p_\calF(y))+\sum_y p_\calF(y)\log\frac{p_\calF(y)}{q(y)}\\
&=H(Y\mid\calF)+D_{\mathrm{KL}}(p_\calF\|q).
\end{align*}
Taking expectation and minimizing over $q$ gives
\begin{align*}
R_{\log}(\calF)=H(Y\mid\calF).
\end{align*}
Likewise,
\begin{align*}
R_{\log}(\calG)=H(Y\mid\calG).
\end{align*}
By definition of conditional mutual information between $Y$ and the additional information $\calG$ given $\calF$,
\begin{align*}
I(Y;\calG\mid\calF)=H(Y\mid\calF)-H(Y\mid\calG).
\end{align*}
Combining proves the claim.
\end{proof}

\begin{theorem}[Squared-loss value, formal version of Theorem~\ref{thm:squared-value:informal}]
\label{thm:squared-value:formal}
Let $Y\in L^2(\Omega;\R^d)$ and let $L(Y,a)=\norm{Y-a}_2^2$. If $\calF\subseteq\calG$, then
\begin{align*}
R_2(\calF)-R_2(\calG)=\E\left[\norm{\E[Y\mid\calG]-\E[Y\mid\calF]}_2^2\right].
\end{align*}
\end{theorem}
\begin{proof}
The optimal $\calF$-measurable predictor is $\E[Y\mid\calF]$, and the optimal $\calG$-measurable predictor is $\E[Y\mid\calG]$. Let
\begin{align*}
\varepsilon=Y-\E[Y\mid\calG],\qquad \delta=\E[Y\mid\calG]-\E[Y\mid\calF].
\end{align*}
Then
\begin{align*}
Y-\E[Y\mid\calF]=\varepsilon+\delta.
\end{align*}
Since $\delta$ is $\calG$-measurable and $\E[\varepsilon\mid\calG]=0$,
\begin{align*}
\E[\ip{\varepsilon}{\delta}\mid\calG]=\ip{\E[\varepsilon\mid\calG]}{\delta}=0.
\end{align*}
Thus
\begin{align*}
\E\norm{Y-\E[Y\mid\calF]}_2^2&=\E\norm{\varepsilon+\delta}_2^2\\
&=\E\norm{\varepsilon}_2^2+\E\norm{\delta}_2^2\\
&=R_2(\calG)+\E\left[\norm{\E[Y\mid\calG]-\E[Y\mid\calF]}_2^2\right].
\end{align*}
Rearranging proves the identity.
\end{proof}

\begin{proposition}[Finite squared-loss split identity]
\label{prop:sse-split:formal}
Let $B=B_0\cup B_1$ be a disjoint split of a finite sample set. With empirical means $\hat\mu_B,\hat\mu_{B_0},\hat\mu_{B_1}$ and $\operatorname{SSE}$ as in Section~\ref{sec:value},
\begin{align*}
\operatorname{SSE}(B)-\operatorname{SSE}(B_0)-\operatorname{SSE}(B_1)=\abs{B_0}\norm{\hat\mu_{B_0}-\hat\mu_B}_2^2+\abs{B_1}\norm{\hat\mu_{B_1}-\hat\mu_B}_2^2.
\end{align*}
\end{proposition}
\begin{proof}
For each child $B_b$,
\begin{align*}
\sum_{j\in B_b}\norm{y_j-\hat\mu_B}_2^2&=\sum_{j\in B_b}\norm{y_j-\hat\mu_{B_b}+\hat\mu_{B_b}-\hat\mu_B}_2^2\\
&=\sum_{j\in B_b}\norm{y_j-\hat\mu_{B_b}}_2^2+\abs{B_b}\norm{\hat\mu_{B_b}-\hat\mu_B}_2^2,
\end{align*}
because
\begin{align*}
\sum_{j\in B_b}(y_j-\hat\mu_{B_b})=0.
\end{align*}
Summing over $b\in\{0,1\}$ gives the result.
\end{proof}

\section{Conservative Memory Proofs}
\label{app:memory-proofs}

\begin{theorem}[Exact associative binding, formal version of Theorem~\ref{thm:exact-binding:informal}]
\label{thm:exact-binding:formal}
Let $K\in\R^{r\times T}$ and $V\in\R^{m\times T}$. The set $\calC=\{W\in\R^{m\times r}:WK=V\}$ is nonempty if and only if $\ker K\subseteq\ker V$. If $\calC$ is nonempty, then
\begin{align*}
\argmin_{W\in\calC}\frac{1}{2}\norm{W-W_0}_F^2=W_0+(V-W_0K)K^\dagger.
\end{align*}
\end{theorem}
\begin{proof}
If $WK=V$ and $a\in\ker K$, then
\begin{align*}
Va=WKa=0.
\end{align*}
Thus $\ker K\subseteq\ker V$.

Assume $\ker K\subseteq\ker V$. Since $K^\dagger K$ is the orthogonal projector onto $(\ker K)^\perp$, the condition implies
\begin{align*}
V=VK^\dagger K.
\end{align*}
Thus $W=VK^\dagger$ satisfies $WK=V$.

Let $R=V-W_0K$ and $W=W_0+\Delta$. The projection problem is
\begin{align*}
\min_\Delta\frac{1}{2}\norm{\Delta}_F^2\quad \text{subject to}\quad \Delta K=R.
\end{align*}
The minimum-norm solution is
\begin{align*}
\Delta^\star=RK^\dagger.
\end{align*}
Therefore
\begin{align*}
W^\star=W_0+(V-W_0K)K^\dagger.
\end{align*}
\end{proof}

\begin{theorem}[Innovation decomposition, formal version of Theorem~\ref{thm:innovation-decomposition:informal}]
\label{thm:innovation-decomposition:formal}
Let $K\in\R^{r\times T}$, $V\in\R^{m\times T}$, $W\in\R^{m\times r}$, $k\in\R^r$, and $v\in\R^m$. Assume $WK=V$. Let $\Pi=KK^\dagger$, $z=(I-\Pi)k$, and $a=K^\dagger k$. Then
\begin{align*}
Wk=Va+Wz.
\end{align*}
If $z=0$, $Wk$ is fixed over all old-feasible $W$. If $z\ne0$, the constraint $Wk=v$ is equivalent over old-feasible $W$ to $Wz=v-Va$.
\end{theorem}
\begin{proof}
The decomposition is
\begin{align*}
k=KK^\dagger k+(I-KK^\dagger)k=Ka+z.
\end{align*}
Multiplying by $W$ gives
\begin{align*}
Wk=WKa+Wz=Va+Wz.
\end{align*}
If $z=0$, then $Wk=Va$ for every $W$ satisfying $WK=V$. If $z\ne0$, then
\begin{align*}
Wk=v\quad \Longleftrightarrow\quad Va+Wz=v\quad \Longleftrightarrow\quad Wz=v-Va.
\end{align*}
\end{proof}

\begin{theorem}[Conflict impossibility, formal version of Theorem~\ref{thm:conflict-impossibility:informal}]
\label{thm:conflict-impossibility:formal}
Let $K\in\R^{r\times T}$, $V\in\R^{m\times T}$, $W_t\in\R^{m\times r}$, $k\in\R^r$, and $v\in\R^m$. Assume $W_tK=V$ and $z=(I-KK^\dagger)k=0$. If $W_t+\Delta$ preserves old evidence, then $(W_t+\Delta)k=W_tk$. Hence $Wk=v$ is compatible with old evidence if and only if $v=W_tk$.
\end{theorem}
\begin{proof}
Preserving old evidence means
\begin{align*}
(W_t+\Delta)K=V=W_tK.
\end{align*}
Thus
\begin{align*}
\Delta K=0.
\end{align*}
If $z=0$, then $k\in\Span(K)$, hence
\begin{align*}
\Delta k=0.
\end{align*}
Therefore
\begin{align*}
(W_t+\Delta)k=W_tk.
\end{align*}
\end{proof}

\begin{theorem}[Hard Conservative Projection Memory]
\label{thm:hard-cpm:formal}
Let $K\in\R^{r\times T}$, $V\in\R^{m\times T}$, $W_t\in\R^{m\times r}$, $k\in\R^r$, and $v\in\R^m$. Assume $W_tK=V$, let $\Pi=KK^\dagger$, $z=(I-\Pi)k\ne0$, and $e=v-W_tk$. Then
\begin{align*}
W_{t+1}=W_t+\frac{ez^\top}{\norm{z}^2}
\end{align*}
preserves $WK=V$, satisfies $Wk=v$, and minimizes $\norm{W-W_t}_F^2/2$ among all memories satisfying both constraints.
\end{theorem}
\begin{proof}
Let $\Delta^\star=ez^\top/\norm{z}^2$. Since $z\perp\Span(K)$,
\begin{align*}
\Delta^\star K=0.
\end{align*}
Thus
\begin{align*}
W_{t+1}K=W_tK=V.
\end{align*}
Also $z^\top k=\norm{z}^2$, so
\begin{align*}
\Delta^\star k=e.
\end{align*}
Hence
\begin{align*}
W_{t+1}k=W_tk+e=v.
\end{align*}
For minimality, any feasible update $\Delta$ satisfies
\begin{align*}
\Delta K=0,\qquad \Delta k=e.
\end{align*}
Since $\Delta K=0$,
\begin{align*}
\Delta k=\Delta(I-\Pi)k=\Delta z.
\end{align*}
Thus the problem reduces to
\begin{align*}
\min_\Delta\frac{1}{2}\norm{\Delta}_F^2\quad \text{subject to}\quad \Delta z=e.
\end{align*}
The minimum-norm solution is $\Delta^\star=ez^\top/\norm{z}^2$.
\end{proof}

\begin{theorem}[No-free-repair lower bound, formal version of Theorem~\ref{thm:no-free-repair:informal}]
\label{thm:no-free-repair:formal}
Let $K\in\R^{r\times T}$, $V\in\R^{m\times T}$, $W_t\in\R^{m\times r}$, $k\in\R^r$, $v\in\R^m$, and $0\le\gamma<1$. Assume $W_tK=V$, let $\Pi=KK^\dagger$, $z=(I-\Pi)k$, and $e=v-W_tk$. Let $W_t+\Delta$ preserve old constraints, so $\Delta K=0$. If $z\ne0$ and
\begin{align*}
\norm{v-(W_t+\Delta)k}\le\gamma\norm{e}
\end{align*}
then
\begin{align*}
\norm{\Delta}_F\ge\frac{(1-\gamma)\norm{e}}{\norm{z}}.
\end{align*}
If $z=0$ and $e\ne0$, no update preserving old constraints can achieve such a contraction.
\end{theorem}
\begin{proof}
Since $\Delta K=0$,
\begin{align*}
\Delta k=\Delta(I-\Pi)k=\Delta z.
\end{align*}
Let
\begin{align*}
e^+=v-(W_t+\Delta)k.
\end{align*}
Then
\begin{align*}
e^+=e-\Delta z.
\end{align*}
Thus
\begin{align*}
\norm{\Delta z}=\norm{e-e^+}\ge\norm{e}-\norm{e^+}\ge(1-\gamma)\norm{e}.
\end{align*}
Also
\begin{align*}
\norm{\Delta z}\le\norm{\Delta}_F\norm{z}.
\end{align*}
If $z\ne0$, division gives the result. If $z=0$, then $\Delta z=0$ and $e^+=e$, so no contraction with $\gamma<1$ is possible when $e\ne0$.
\end{proof}

\begin{theorem}[Optimal contraction, formal version of Theorem~\ref{thm:optimal-contraction:informal}]
\label{thm:optimal-contraction:formal}
Let $K\in\R^{r\times T}$, $V\in\R^{m\times T}$, $W_t\in\R^{m\times r}$, $k\in\R^r$, $v\in\R^m$, and $0\le\gamma\le1$. Assume $W_tK=V$, let $\Pi=KK^\dagger$, $z=(I-\Pi)k\ne0$, and $e=v-W_tk$. Among all updates satisfying
\begin{align*}
\Delta K=0,\qquad v-(W_t+\Delta)k=\gamma e,
\end{align*}
the unique minimum-Frobenius-norm update is
\begin{align*}
\Delta^\star=(1-\gamma)\frac{ez^\top}{\norm{z}^2}.
\end{align*}
Its norm is
\begin{align*}
\norm{\Delta^\star}_F=\frac{(1-\gamma)\norm{e}}{\norm{z}}.
\end{align*}
\end{theorem}
\begin{proof}
The residual constraint is
\begin{align*}
e-\Delta k=\gamma e.
\end{align*}
Since $\Delta K=0$, $\Delta k=\Delta z$. Thus
\begin{align*}
\Delta z=(1-\gamma)e.
\end{align*}
The minimum-norm solution of this linear constraint is
\begin{align*}
\Delta^\star=(1-\gamma)\frac{ez^\top}{\norm{z}^2}.
\end{align*}
It preserves old constraints because $z^\top K=0$:
\begin{align*}
\Delta^\star K=0.
\end{align*}
It achieves the target because
\begin{align*}
\Delta^\star z=(1-\gamma)e.
\end{align*}
The norm is
\begin{align*}
\norm{\Delta^\star}_F=(1-\gamma)\frac{\norm{e}\norm{z}}{\norm{z}^2}=\frac{(1-\gamma)\norm{e}}{\norm{z}}.
\end{align*}
\end{proof}

\begin{theorem}[Sequential hard CPM equals batch projection, formal version of Theorem~\ref{thm:sequential-batch:informal}]
\label{thm:sequential-batch:formal}
Let $(k_t,v_t)_{t=1}^T$ be compatible evidence, and let $K_t=[k_1,\ldots,k_t]$ and $V_t=[v_1,\ldots,v_t]$. Initialize $W_1=W_0$. At step $t$, let $\Pi_{t-1}=K_{t-1}K_{t-1}^\dagger$ and $z_t=(I-\Pi_{t-1})k_t$. If $z_t=0$, make no update. If $z_t\ne0$, apply hard CPM:
\begin{align*}
W_{t+1}=W_t+\frac{(v_t-W_tk_t)z_t^\top}{\norm{z_t}^2}.
\end{align*}
Then
\begin{align*}
W_{T+1}=\argmin_{W:WK_T=V_T}\frac{1}{2}\norm{W-W_0}_F^2.
\end{align*}
\end{theorem}
\begin{proof}
Let
\begin{align*}
\calC_t=\{W:WK_t=V_t\},\qquad \calT_t=\{H:HK_t=0\}.
\end{align*}
We prove by induction that $W_{t+1}$ is the Euclidean projection of $W_0$ onto $\calC_t$. For $t=0$, this is $W_1=W_0$.

Assume $W_t=\operatorname{Proj}_{\calC_{t-1}}(W_0)$. Thus $W_t\in\calC_{t-1}$ and
\begin{align*}
\ip{H}{W_t-W_0}_F=0\qquad \forall H\in\calT_{t-1}.
\end{align*}
If $z_t=0$, compatibility implies the new constraint is already implied by $\calC_{t-1}$, hence $\calC_t=\calC_{t-1}$ and no update preserves the projection.

If $z_t\ne0$, hard CPM gives $W_{t+1}\in\calC_t$ by Theorem~\ref{thm:hard-cpm:formal}. Let $H\in\calT_t$. Then $H\in\calT_{t-1}$ and $Hk_t=0$. Since $\Pi_{t-1}k_t\in\Span(K_{t-1})$,
\begin{align*}
Hz_t=Hk_t-H\Pi_{t-1}k_t=0.
\end{align*}
Let $\Delta_t=W_{t+1}-W_t$. Then
\begin{align*}
\ip{H}{\Delta_t}_F=\ip{H}{(v_t-W_tk_t)z_t^\top}_F\frac{1}{\norm{z_t}^2}=\frac{(v_t-W_tk_t)^\top Hz_t}{\norm{z_t}^2}=0.
\end{align*}
Therefore
\begin{align*}
\ip{H}{W_{t+1}-W_0}_F=\ip{H}{W_t-W_0}_F+\ip{H}{\Delta_t}_F=0.
\end{align*}
The Euclidean projection characterization onto the affine set $\calC_t$ gives $W_{t+1}=\operatorname{Proj}_{\calC_t}(W_0)$.
\end{proof}

\begin{theorem}[Soft Conservative Projection Memory, formal version of Theorem~\ref{thm:soft-cpm:informal}]
\label{thm:soft-cpm:formal}
Let $K\in\R^{r\times T}$, $V\in\R^{m\times T}$, $W_t\in\R^{m\times r}$, $k\in\R^r$, $v\in\R^m$, and $\eta>0$. Assume $W_tK=V$, let $\Pi=KK^\dagger$, $z=(I-\Pi)k\ne0$, and $e=v-W_tk$. The unique solution of
\begin{align*}
\argmin_{W:WK=V}\left\{\frac{1}{2}\norm{W-W_t}_F^2+\frac{\eta}{2}\norm{Wk-v}^2\right\}
\end{align*}
is
\begin{align*}
W_{t+1}=W_t+\frac{\eta}{1+\eta\norm{z}^2}ez^\top.
\end{align*}
Moreover,
\begin{align*}
W_{t+1}K=V,\qquad v-W_{t+1}k=\frac{1}{1+\eta\norm{z}^2}e.
\end{align*}
\end{theorem}
\begin{proof}
Write $W=W_t+\Delta$. The constraint $WK=V$ is equivalent to
\begin{align*}
\Delta K=0.
\end{align*}
Since $\Delta K=0$,
\begin{align*}
\Delta k=\Delta(I-\Pi)k=\Delta z.
\end{align*}
The objective becomes
\begin{align*}
\min_{\Delta:\Delta K=0}\frac{1}{2}\norm{\Delta}_F^2+\frac{\eta}{2}\norm{\Delta z-e}^2.
\end{align*}
The stationarity equation in the feasible subspace is
\begin{align*}
\Delta+\eta(\Delta z-e)z^\top=0.
\end{align*}
Let $\Delta=\alpha ez^\top$. Substitution gives
\begin{align*}
\alpha ez^\top+\eta(\alpha e\norm{z}^2-e)z^\top=0.
\end{align*}
Thus
\begin{align*}
\alpha(1+\eta\norm{z}^2)=\eta.
\end{align*}
Therefore
\begin{align*}
\Delta=\frac{\eta}{1+\eta\norm{z}^2}ez^\top.
\end{align*}
Since $z^\top K=0$, $W_{t+1}K=V$. The residual is
\begin{align*}
v-W_{t+1}k&=e-\Delta k\\
&=e-\Delta z\\
&=e-\frac{\eta\norm{z}^2}{1+\eta\norm{z}^2}e\\
&=\frac{1}{1+\eta\norm{z}^2}e.
\end{align*}
\end{proof}

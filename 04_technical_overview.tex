\section{Proof Sketches}
\label{sec:technical-overview}

This section collects the main proof ideas in the order used by the paper: runtime excess risk decomposes into $\Delta_{\mathrm{repair}}$ and $\Delta_{\mathrm{refine}}$; repair-only learners have linear regret against refined oracles; useful refinements are exactly those that change the Bayes action; minimal refinement adds the missing distinction and nothing else; useful distinctions have exact value under standard losses; and guarded DGM updates decrease the local penalized objective. The measure-theoretic notation, conditional-expectation facts, and decision-theoretic Bayes-risk framing are standard \citep{billingsley1995probability,kallenberg2002foundations,blackwell1953equivalent,berger1985statistical}. Full formal versions appear in Appendices~\ref{app:refinement-proofs}, \ref{app:value-proofs}, and \ref{app:algorithm-bridge-proofs}.

\begin{theorem}[Runtime error decomposition, proof sketch for Theorem~\ref{thm:error-decomposition}]
\label{thm:error-decomposition:sketch}
The statement is Theorem~\ref{thm:error-decomposition}.
\end{theorem}
\begin{proof}[Proof sketch]
Add and subtract the Bayes risk under the current information:
\begin{align*}
R_L(f_t)-R_L(\Pi^\star)
=
\Delta_{\mathrm{repair}}(f_t,\Pi_t)
+
\Delta_{\mathrm{refine}}(\Pi_t,\Pi^\star).
\end{align*}
The first term is nonnegative because $f_t$ is one of the $\Pi_t$-measurable predictors over which $R_L(\Pi_t)$ optimizes. The second is nonnegative because a refinement enlarges the admissible predictor class.
\end{proof}

\begin{theorem}[Repair-only regret, proof sketch for Theorem~\ref{thm:repair-only-regret}]
\label{thm:repair-only-regret:sketch}
The statement is Theorem~\ref{thm:repair-only-regret}.
\end{theorem}
\begin{proof}[Proof sketch]
Condition on the history before round $t$. A repair-only learner's prediction is $\Pi$-measurable, so its conditional expected loss is at least $R_L(\Pi)$ by definition of Bayes risk under the partition. Summing over rounds gives cumulative expected loss at least $TR_L(\Pi)$. A refined oracle using $\Pi^\star$ pays $TR_L(\Pi^\star)$, so the regret lower bound follows by subtraction.
\end{proof}

\begin{theorem}[Loss-relevant refinement, proof sketch for Theorem~\ref{thm:loss-relevant-refinement:formal}]
\label{thm:loss-relevant-refinement:sketch}
The statement is Theorem~\ref{thm:loss-relevant-refinement:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
If the $\calG$-Bayes action is already $\calF$-measurable, then it is feasible under the old information and attains the refined risk, so the refinement gap is zero. Conversely, if the old and refined risks are equal, any $\calF$-Bayes action is also $\calG$-optimal. Almost sure uniqueness of the $\calG$-Bayes action forces equality with the refined Bayes action, so the refined action has an $\calF$-measurable version.
\end{proof}

\begin{theorem}[Contradiction, proof sketch for Theorem~\ref{thm:contradiction:formal}]
\label{thm:contradiction:sketch}
The statement is Corollary~\ref{thm:contradiction:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
If $f$ is $\calP$-measurable and $\omega,\omega'$ lie in the same atom, then
\begin{align*}
f(\omega)=f(\omega').
\end{align*}
If $f(\omega)=y(\omega)$ and $f(\omega')=y(\omega')$, then
\begin{align*}
y(\omega)=y(\omega'),
\end{align*}
contradicting the hypothesis.
\end{proof}

\begin{theorem}[Minimal finite refinement, proof sketch for Theorem~\ref{thm:minimal-refinement:formal}]
\label{thm:minimal-refinement:sketch}
The statement is Theorem~\ref{thm:minimal-refinement:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
Every atom of $\calP_D$ lies inside an atom of $\calP$, so $\calP_D$ refines $\calP$. Also
\begin{align*}
D=\bigcup_{B\in\calP}(B\cap D),
\end{align*}
so $D$ is $\calP_D$-measurable. If $\calQ$ refines $\calP$ and makes $D$ measurable, then every atom $Q\in\calQ$ satisfies
\begin{align*}
Q\subseteq B
\end{align*}
for some $B\in\calP$, and also either $Q\subseteq D$ or $Q\subseteq D^c$. Hence
\begin{align*}
Q\subseteq B\cap D\qquad \text{or}\qquad Q\subseteq B\cap D^c.
\end{align*}
Thus $\calQ$ refines $\calP_D$.
\end{proof}
\begin{theorem}[Minimal sigma-algebra refinement, proof sketch for Theorem~\ref{thm:sigma-refinement:formal}]
\label{thm:sigma-refinement:sketch}
The statement is Theorem~\ref{thm:sigma-refinement:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
By definition,
\begin{align*}
\calF\vee\sigma(U)=\sigma(\calF\cup\sigma(U)).
\end{align*}
This sigma-algebra contains $\calF$ and makes $U$ measurable. If $\calG$ also contains $\calF$ and makes $U$ measurable, then
\begin{align*}
\calF\cup\sigma(U)\subseteq\calG.
\end{align*}
Since $\calG$ is a sigma-algebra,
\begin{align*}
\sigma(\calF\cup\sigma(U))\subseteq\calG.
\end{align*}
\end{proof}

\begin{theorem}[Risk monotonicity, proof sketch for Theorem~\ref{thm:risk-monotonicity:formal}]
\label{thm:risk-monotonicity:sketch}
The statement is Theorem~\ref{thm:risk-monotonicity:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
If $\calF\subseteq\calG$, then every $\calF$-measurable action is $\calG$-measurable. Therefore
\begin{align*}
\{a:a\text{ is }\calF\text{-measurable}\}\subseteq\{a:a\text{ is }\calG\text{-measurable}\}.
\end{align*}
Taking the infimum of the same expected loss over the larger set gives
\begin{align*}
R_L(\calG)\le R_L(\calF).
\end{align*}
\end{proof}

\begin{theorem}[Log-loss value, proof sketch for Theorem~\ref{thm:log-value:formal}]
\label{thm:log-value:sketch}
The statement is Theorem~\ref{thm:log-value:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
For any $\calF$-measurable conditional distribution $q$, the standard log-loss decomposition gives
\begin{align*}
\E[-\log q(Y)]=H(Y\mid\calF)+\E\left[D_{\mathrm{KL}}(\mathbb P(Y\in\cdot\mid\calF)\|q(\cdot))\right].
\end{align*}
The minimizer is $q^\star=\mathbb P(Y\in\cdot\mid\calF)$, so $R_{\log}(\calF)=H(Y\mid\calF)$. Similarly $R_{\log}(\calG)=H(Y\mid\calG)$. Therefore
\begin{align*}
R_{\log}(\calF)-R_{\log}(\calG)=H(Y\mid\calF)-H(Y\mid\calG)=I(Y;\calG\mid\calF).
\end{align*}
\end{proof}

\begin{theorem}[Squared-loss value, proof sketch for Theorem~\ref{thm:squared-value:formal}]
\label{thm:squared-value:sketch}
The statement is Theorem~\ref{thm:squared-value:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
The optimal $\calF$-predictor is $\E[Y\mid\calF]$, the standard $L^2$ projection. Since $\calF\subseteq\calG$,
\begin{align*}
Y-\E[Y\mid\calF]=\left(Y-\E[Y\mid\calG]\right)+\left(\E[Y\mid\calG]-\E[Y\mid\calF]\right).
\end{align*}
The two terms are orthogonal because the first has conditional mean zero given $\calG$ and the second is $\calG$-measurable. Taking squared $\ell_2$ norms and expectations gives the identity.
\end{proof}

\begin{theorem}[Anchor edge separation, proof sketch for Theorem~\ref{thm:anchor-edge:formal}]
\label{thm:anchor-edge:sketch}
The statement is Theorem~\ref{thm:anchor-edge:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
With $u=(a_i-a_j)/\norm{a_i-a_j}_2$ and $b=u^\top(a_i+a_j)/2$,
\begin{align*}
u^\top a_i-b&=\frac{1}{2}u^\top(a_i-a_j)=\frac{1}{2}\norm{a_i-a_j}_2,\\
u^\top a_j-b&=-\frac{1}{2}u^\top(a_i-a_j)=-\frac{1}{2}\norm{a_i-a_j}_2.
\end{align*}
\end{proof}

\begin{theorem}[DGM graph partition, proof sketch for Theorem~\ref{thm:dgm-graph-partition:formal}]
\label{thm:dgm-graph-partition:sketch}
The statement is Theorem~\ref{thm:dgm-graph-partition:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
The graph signature records the value of every binary edge generator. Equality of signatures is an equivalence relation, and its classes are exactly the nonempty common level sets of those generators. Adding a new edge appends one binary coordinate to the signature, so each old atom is either left intact or cut by the two events $g^{-1}(1)$ and $g^{-1}(0)$.
\end{proof}

\begin{theorem}[Hard-route measurability, proof sketch for Theorem~\ref{thm:hard-dgm-measurable:formal}]
\label{thm:hard-dgm-measurable:sketch}
The statement is Theorem~\ref{thm:hard-dgm-measurable:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
Edge eligibility is determined by finitely many edge-bit events, so it is measurable with respect to $\calF_t^{\mathrm{edge}}$. The anchor-order sigma-algebra generated by all pairwise anchor comparisons determines the nearest anchor in any eligible subset under deterministic tie-breaking. The exact hard selected index is a finite composition of these two finite decisions, so it is measurable with respect to their join. Movable centroids can still be used for the soft retrieval path, but they are not part of this formal information structure.
\end{proof}

\begin{theorem}[Soft routing hard limit, proof sketch for Theorem~\ref{thm:soft-routing-hard-limit:formal}]
\label{thm:soft-routing-hard-limit:sketch}
The statement is Theorem~\ref{thm:soft-routing-hard-limit:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
The sparse soft read can only assign mass to candidates it retrieved. Therefore, if the exact hard-route concept is omitted from the candidate set, no temperature limit can recover that route. Under candidate recall, the remaining argument is a finite softmax and gate-sharpening calculation.
For ordinary nearest-center routing, every nonnearest candidate has unnormalized weight smaller by at least $\exp(-\delta/T)$, so the total nonnearest mass is bounded by a finite candidate-count factor times this exponential. For a structural edge gate with uniformly positive weight, a wrong-side signed margin $-\delta_e$ contributes a log-sigmoid term bounded by $-\delta_e/\tau_e$. Degree normalization rescales the exponent but does not remove exponential decay under a uniform degree bound. After normalizing against an eligible candidate with compatibility bounded below, the wrong-side mass vanishes if the wrong-side/reference distance-score ratio is bounded, or more generally if the joint limit satisfies $\beta_{\min}\delta_e/(\sqrt{D_{\max}}\tau_e)-\Delta_d/T\to+\infty$.
\end{proof}

\begin{theorem}[Refine-or-repair descent, proof sketch for Theorem~\ref{thm:refine-repair-descent:formal}]
\label{thm:refine-repair-descent:sketch}
The statement is Theorem~\ref{thm:refine-repair-descent:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
Guarded repair proposes local memory or centroid changes while keeping anchors and edges fixed, then commits only candidates that do not increase empirical loss on the fixed scoring buffer $\calB_t^{\mathrm{score}}$. The identity candidate is always feasible. The concept and edge penalties therefore stay fixed, so the local objective $J_{\calB_t^{\mathrm{score}}}$ does not increase. A refinement that creates one anchored concept and accepted edge set $E_{\mathrm{new}}$ changes $J_{\calB_t^{\mathrm{score}}}$ by the empirical-risk decrease on the scoring buffer minus $\lambda_{\mathrm{concept}}+\lambda_{\mathrm{edge}}\abs{E_{\mathrm{new}}}$, where the concept penalty includes the induced anchor-order bookkeeping. This is exactly the penalized refinement gain in \eqref{eq:refine-gain}. The acceptance test requires this net decrease to exceed the empirical repair gain plus $\tau+2\epsilon_m$.
\end{proof}

\begin{theorem}[Safe refinement, proof sketch for Theorem~\ref{thm:safe-refinement:formal}]
\label{thm:safe-refinement:sketch}
The statement is Theorem~\ref{thm:safe-refinement:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
At each time $t$, condition on the filtration containing the past, current DGM state, and proposal buffer. The predictable finite candidate class is then fixed, while the scoring buffer remains fresh or otherwise unused by prior proposal and state-selection decisions. Hoeffding's inequality bounds the deviation between empirical and population risk for the old state and for each candidate-refined state \citep{hoeffding1963probability}. A union bound over the finite class gives failure probability of order $\delta/t^2$ after using the displayed $\epsilon_t$, and a second union bound over all times is summable. Each raw gain is a difference of two risks, so its empirical-population deviation is at most twice the risk deviation. If the empirical raw gain exceeds complexity cost plus $\tau+2\epsilon_t$, the population raw gain exceeds cost plus $\tau$ on this high-probability event.
\end{proof}

\begin{corollary}[Bounded LM score, proof sketch for Corollary~\ref{cor:lm-clipped-refinement:formal}]
\label{cor:lm-clipped-refinement:sketch}
The statement is Corollary~\ref{cor:lm-clipped-refinement:informal}.
\end{corollary}
\begin{proof}[Proof sketch]
Clipping and normalizing next-token negative log likelihood gives a score in $[0,1]$. The adaptive refinement theorem therefore applies directly to the clipped scoring rule. The conclusion is intentionally about clipped population gain, not raw unbounded cross-entropy.
\end{proof}

\begin{theorem}[Anchor-stable DGM information structure, proof sketch for Theorem~\ref{thm:anchor-stable:formal}]
\label{thm:anchor-stable:sketch}
The statement is Theorem~\ref{thm:anchor-stable:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
Every edge generator $g_e(h)=\one\{u_e^\top h>b_e\}$ is a deterministic function of the two frozen endpoint anchors. A repair step changes neither endpoint anchor, so it changes neither $u_e$ nor $b_e$. It also leaves every anchor-distance comparison event unchanged. Therefore the anchor-generated DGM information structure is unchanged, even though the retrieval centroids and local memories may be repaired.
\end{proof}

\begin{corollary}[Finite refinements, proof sketch for Corollary~\ref{cor:finite-refinements:formal}]
\label{cor:finite-refinements:sketch}
The statement is Corollary~\ref{cor:finite-refinements:informal}.
\end{corollary}
\begin{proof}[Proof sketch]
After $m$ accepted refinements a common potential $\Phi$ has decreased by at least $m\tau$. Since the potential is nonnegative, $m\tau\le \Phi(\calS_0)$. This argument requires a common potential; changing local scoring buffers do not by themselves define a global finiteness bound.
\end{proof}

\begin{proposition}[Tree as generalized hard-DGM, proof sketch for Proposition~\ref{prop:tree-as-dgm:formal}]
\label{prop:tree-as-dgm:sketch}
The statement is Proposition~\ref{prop:tree-as-dgm:informal}.
\end{proposition}
\begin{proof}[Proof sketch]
Make one concept for each tree leaf and attach the root-to-leaf binary tests as hard eligibility constraints. Every input satisfies exactly the path constraints of the leaf reached by the tree, so the generalized hard route selects the same memory and returns the same prediction.
\end{proof}

\begin{theorem}[DGM data structure, proof sketch for Theorem~\ref{thm:dgm-complexity:formal}]
\label{thm:dgm-complexity:sketch}
The statement is Theorem~\ref{thm:dgm-complexity:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
A prediction computes centroid and anchor distances from $h$ to $N_t$ concepts in $\R^d$, costing
\begin{align*}
O(N_td).
\end{align*}
It then evaluates $Q(h)$ selected incident edge gates and $\abs{N(h)}$ selected local memories. Dense linear edge gates cost $O(Q(h)d)$, so this part costs $O(Q(h)d+\abs{N(h)}d+C_{\mathrm{local}})$ including sparse routing weights. Guarded repair adds the cost of scoring proposed local candidates. Creating one anchor-centroid pair stores two vectors in $\R^d$, still $O(d)$. Adding $q$ distinction edges stores $q$ vectors and thresholds, costing
\begin{align*}
O(qd).
\end{align*}
For mean memories, each concept stores $a_j,c_j\in\R^d$ and $r_j\in\R^{d_e}$, while each edge stores one distinction vector in $\R^d$. Hence memory is
\begin{align*}
O(N_t(2d+d_e)+\abs{E_t}d).
\end{align*}
\end{proof}

\begin{theorem}[DGM layer cost, proof sketch for Theorem~\ref{thm:dgm-layer-cost:formal}]
\label{thm:dgm-layer-cost:sketch}
The statement is Theorem~\ref{thm:dgm-layer-cost:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
Computing $q=Qh$ costs $O(dd_q)$. Retrieval costs $C_{\mathrm{ret}}(N_t,d_q)$ by definition. Dense edge-gate evaluation over selected incident edges costs $O(Q_N d_q)$. For each selected concept, a rank-$r$ adapter computes
\begin{align*}
W_jKh.
\end{align*}
The matrix-vector product with one local adapter costs $O(dr)$, so $\abs{N}$ selected adapters cost
\begin{align*}
O(\abs{N}dr).
\end{align*}
The detached write path repeats query, retrieval, and edge-gate evaluation, then updates only selected local states. A rank-one local repair or low-rank sufficient-statistic update costs $O(dr)$ per selected memory, with $O(d_q)$ additional concept bookkeeping. Hence write overhead is
\begin{align*}
O(dd_q+C_{\mathrm{ret}}(N_t,d_q)+Q_N d_q+\abs{N}(d_q+dr)).
\end{align*}
Because the write path is detached, backpropagation does not need intermediate copies of $\calS_t$ for all $t$. It stores the selected read computation used in the forward loss.
\end{proof}

\begin{theorem}[One-hot memory, proof sketch for Theorem~\ref{thm:one-hot-measurable:formal}]
\label{thm:one-hot-measurable:sketch}
The statement is Theorem~\ref{thm:one-hot-measurable:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
If $\omega,\omega'\in B_i$, then both one-hot features equal $e_i$, so
\begin{align*}
W\phi_\calP(\omega)=We_i=W\phi_\calP(\omega').
\end{align*}
Thus $W\phi_\calP$ is constant on atoms. Conversely, if $f$ is constant on each $B_i$, put that constant value in column $i$ of $W$. Then
\begin{align*}
f(\omega)=W\phi_\calP(\omega)
\end{align*}
for every $\omega$.
\end{proof}

\begin{theorem}[CPM conflict, proof sketch for Theorem~\ref{thm:cpm-conflict-distinction:formal}]
\label{thm:cpm-conflict-distinction:sketch}
The statement is Theorem~\ref{thm:cpm-conflict-distinction:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
If $\omega,\omega'\in B_i$, then
\begin{align*}
\phi_\calP(\omega)=\phi_\calP(\omega')=e_i.
\end{align*}
After committing $We_i=v(\omega)$, the later evidence $We_i=v(\omega')$ has the same key. Its innovation relative to the old key span is
\begin{align*}
z=(I-e_ie_i^\top)e_i=0.
\end{align*}
Its residual is
\begin{align*}
e=v(\omega')-We_i=v(\omega')-v(\omega)\ne0.
\end{align*}
\end{proof}

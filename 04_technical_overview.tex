\section{Technical Overview}
\label{sec:technical-overview}

This section gives the proof ideas in the same order as the main text. The measure-theoretic notation, conditional-expectation facts, and decision-theoretic Bayes-risk framing are standard \citep{billingsley1995probability,kallenberg2002foundations,blackwell1953equivalent,berger1985statistical}. Full formal versions appear in Appendices~\ref{app:refinement-proofs}, \ref{app:value-proofs}, and \ref{app:algorithm-bridge-proofs}.

\begin{theorem}[Minimal finite refinement, proof sketch for Theorem~\ref{thm:minimal-refinement:formal}]
\label{thm:minimal-refinement:sketch}
The main-text statement is Theorem~\ref{thm:minimal-refinement:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
Every atom of $\calP_E$ lies inside an atom of $\calP$, so $\calP_E$ refines $\calP$. Also
\begin{align*}
E=\bigcup_{B\in\calP}(B\cap E),
\end{align*}
so $E$ is $\calP_E$-measurable. If $\calQ$ refines $\calP$ and makes $E$ measurable, then every atom $Q\in\calQ$ satisfies
\begin{align*}
Q\subseteq B
\end{align*}
for some $B\in\calP$, and also either $Q\subseteq E$ or $Q\subseteq E^c$. Hence
\begin{align*}
Q\subseteq B\cap E\qquad \text{or}\qquad Q\subseteq B\cap E^c.
\end{align*}
Thus $\calQ$ refines $\calP_E$.
\end{proof}
\begin{theorem}[Minimal sigma-algebra refinement, proof sketch for Theorem~\ref{thm:sigma-refinement:formal}]
\label{thm:sigma-refinement:sketch}
The main-text statement is Theorem~\ref{thm:sigma-refinement:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
By definition,
\begin{align*}
\calF\vee\sigma(Z)=\sigma(\calF\cup\sigma(Z)).
\end{align*}
This sigma-algebra contains $\calF$ and makes $Z$ measurable. If $\calG$ also contains $\calF$ and makes $Z$ measurable, then
\begin{align*}
\calF\cup\sigma(Z)\subseteq\calG.
\end{align*}
Since $\calG$ is a sigma-algebra,
\begin{align*}
\sigma(\calF\cup\sigma(Z))\subseteq\calG.
\end{align*}
\end{proof}

\begin{theorem}[Contradiction, proof sketch for Theorem~\ref{thm:contradiction:formal}]
\label{thm:contradiction:sketch}
The main-text statement is Theorem~\ref{thm:contradiction:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
If $f$ is $\calP$-measurable and $\omega,\omega'$ lie in the same atom, then
\begin{align*}
f(\omega)=f(\omega').
\end{align*}
If $f(\omega)=y(\omega)$ and $f(\omega')=y(\omega')$, then
\begin{align*}
y(\omega)=y(\omega'),
\end{align*}
contradicting the hypothesis.
\end{proof}

\begin{theorem}[Risk monotonicity, proof sketch for Theorem~\ref{thm:risk-monotonicity:formal}]
\label{thm:risk-monotonicity:sketch}
The main-text statement is Theorem~\ref{thm:risk-monotonicity:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
If $\calF\subseteq\calG$, then every $\calF$-measurable action is $\calG$-measurable. Therefore
\begin{align*}
\{a:a\text{ is }\calF\text{-measurable}\}\subseteq\{a:a\text{ is }\calG\text{-measurable}\}.
\end{align*}
Taking the infimum of the same expected loss over the larger set gives
\begin{align*}
R_L(\calG)\le R_L(\calF).
\end{align*}
\end{proof}

\begin{theorem}[Log-loss value, proof sketch for Theorem~\ref{thm:log-value:formal}]
\label{thm:log-value:sketch}
The main-text statement is Theorem~\ref{thm:log-value:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
For any $\calF$-measurable conditional distribution $q$, the standard log-loss decomposition gives
\begin{align*}
\E[-\log q(Y)]=H(Y\mid\calF)+\E\left[D_{\mathrm{KL}}(\mathbb P(Y\in\cdot\mid\calF)\|q(\cdot))\right].
\end{align*}
The minimizer is $q^\star=\mathbb P(Y\in\cdot\mid\calF)$, so $R_{\log}(\calF)=H(Y\mid\calF)$. Similarly $R_{\log}(\calG)=H(Y\mid\calG)$. Therefore
\begin{align*}
R_{\log}(\calF)-R_{\log}(\calG)=H(Y\mid\calF)-H(Y\mid\calG)=I(Y;\calG\mid\calF).
\end{align*}
\end{proof}

\begin{theorem}[Squared-loss value, proof sketch for Theorem~\ref{thm:squared-value:formal}]
\label{thm:squared-value:sketch}
The main-text statement is Theorem~\ref{thm:squared-value:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
The optimal $\calF$-predictor is $\E[Y\mid\calF]$, the standard $L^2$ projection. Since $\calF\subseteq\calG$,
\begin{align*}
Y-\E[Y\mid\calF]=\left(Y-\E[Y\mid\calG]\right)+\left(\E[Y\mid\calG]-\E[Y\mid\calF]\right).
\end{align*}
The two terms are orthogonal because the first has conditional mean zero given $\calG$ and the second is $\calG$-measurable. Taking squared $\ell_2$ norms and expectations gives the identity.
\end{proof}

\begin{theorem}[Prototype edge separation, proof sketch for Theorem~\ref{thm:prototype-edge:formal}]
\label{thm:prototype-edge:sketch}
The main-text statement is Theorem~\ref{thm:prototype-edge:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
With $u=(a_i-a_j)/\norm{a_i-a_j}_2$ and $b=u^\top(a_i+a_j)/2$,
\begin{align*}
u^\top a_i-b&=\frac{1}{2}u^\top(a_i-a_j)=\frac{1}{2}\norm{a_i-a_j}_2,\\
u^\top a_j-b&=-\frac{1}{2}u^\top(a_i-a_j)=-\frac{1}{2}\norm{a_i-a_j}_2.
\end{align*}
\end{proof}

\begin{theorem}[DGM graph partition, proof sketch for Theorem~\ref{thm:dgm-graph-partition:formal}]
\label{thm:dgm-graph-partition:sketch}
The main-text statement is Theorem~\ref{thm:dgm-graph-partition:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
The graph signature records the value of every binary edge generator. Equality of signatures is an equivalence relation, and its classes are exactly the nonempty common level sets of those generators. Adding a new edge appends one binary coordinate to the signature, so each old atom is either left intact or cut by the two events $g^{-1}(1)$ and $g^{-1}(0)$.
\end{proof}

\begin{theorem}[Hard-DGM measurability, proof sketch for Theorem~\ref{thm:hard-dgm-measurable:formal}]
\label{thm:hard-dgm-measurable:sketch}
The main-text statement is Theorem~\ref{thm:hard-dgm-measurable:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
Edge eligibility is determined by finitely many edge-bit events, so it is measurable with respect to $\calF_t^{\mathrm{edge}}$. The anchor-order sigma-algebra generated by all pairwise anchor comparisons determines the nearest anchor in any eligible subset under deterministic tie-breaking. The exact hard selected index is a finite composition of these two finite decisions, so it is measurable with respect to their join. Movable centroids can still be used for the soft retrieval path, but they are not part of this formal information structure.
\end{proof}

\begin{theorem}[Soft routing hard limit, proof sketch for Theorem~\ref{thm:soft-routing-hard-limit:formal}]
\label{thm:soft-routing-hard-limit:sketch}
The main-text statement is Theorem~\ref{thm:soft-routing-hard-limit:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
For ordinary nearest-prototype routing, every nonnearest candidate has unnormalized weight smaller by at least $\exp(-\delta/T)$, so the total nonnearest mass is bounded by a finite candidate-count factor times this exponential. For a structural edge gate with uniformly positive weight, a wrong-side signed margin $-\delta_e$ contributes a log-sigmoid term bounded by $-\delta_e/\tau_e$. Degree normalization rescales the exponent but does not remove exponential decay under a uniform degree bound. After normalizing against an eligible candidate with compatibility bounded below and bounded distance-score ratio, the wrong-side mass vanishes as the edge temperature tends to zero.
\end{proof}

\begin{theorem}[Refine-or-repair descent, proof sketch for Theorem~\ref{thm:refine-repair-descent:formal}]
\label{thm:refine-repair-descent:sketch}
The main-text statement is Theorem~\ref{thm:refine-repair-descent:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
Monotone repair changes only local memories and retrieval centroids while keeping anchors and edges fixed, and by definition does not increase the empirical loss on the fixed scoring buffer $\calB_t^{\mathrm{score}}$. The prototype and edge penalties therefore stay fixed, so the local objective $J_{\calB_t^{\mathrm{score}}}$ does not increase. A refinement that creates one anchored concept and $q$ edges changes $J_{\calB_t^{\mathrm{score}}}$ by the empirical-risk decrease on the scoring buffer minus $\lambda_{\mathrm{proto}}+\lambda_{\mathrm{edge}}q$. The acceptance test requires this net decrease to exceed $\tau+2\epsilon_m$.
\end{proof}

\begin{theorem}[Safe refinement, proof sketch for Theorem~\ref{thm:safe-refinement:formal}]
\label{thm:safe-refinement:sketch}
The main-text statement is Theorem~\ref{thm:safe-refinement:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
Conditioning on a finite candidate class fixed before scoring, or generated from a proposal buffer independent of the scoring buffer, Hoeffding's inequality bounds the deviation between empirical and population risk for the old state and for each candidate-refined state \citep{hoeffding1963probability}. A union bound over this finite collection gives the usual finite-class validation guarantee \citep{devroye1996probabilistic}. Each gain is a difference of two risks, so its empirical-population deviation is at most twice the risk deviation. If the empirical gain exceeds cost plus $\tau+2\epsilon_m$, the population gain exceeds cost plus $\tau$ on this high-probability event.
\end{proof}

\begin{theorem}[Anchor-stable DGM information structure, proof sketch for Theorem~\ref{thm:anchor-stable:formal}]
\label{thm:anchor-stable:sketch}
The main-text statement is Theorem~\ref{thm:anchor-stable:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
Every edge generator $g_e(h)=\one\{u_e^\top h>b_e\}$ is a deterministic function of the two frozen endpoint anchors. A repair step changes neither endpoint anchor, so it changes neither $u_e$ nor $b_e$. It also leaves every anchor-distance comparison event unchanged. Therefore the anchor-generated DGM information structure is unchanged, even though the retrieval centroids and local memories may be repaired.
\end{proof}

\begin{corollary}[Finite refinements, proof sketch for Corollary~\ref{cor:finite-refinements:formal}]
\label{cor:finite-refinements:sketch}
The main-text statement is Corollary~\ref{cor:finite-refinements:informal}.
\end{corollary}
\begin{proof}[Proof sketch]
After $m$ accepted refinements a common potential $\Phi$ has decreased by at least $m\tau$. Since the potential is nonnegative, $m\tau\le \Phi(\calS_0)$. This argument requires a common potential; changing local scoring buffers do not by themselves give a global finiteness bound.
\end{proof}

\begin{proposition}[Tree as generalized hard-DGM, proof sketch for Proposition~\ref{prop:tree-as-dgm:formal}]
\label{prop:tree-as-dgm:sketch}
The main-text statement is Proposition~\ref{prop:tree-as-dgm:informal}.
\end{proposition}
\begin{proof}[Proof sketch]
Make one prototype for each tree leaf and attach the root-to-leaf binary tests as hard eligibility constraints. Every input satisfies exactly the path constraints of the leaf reached by the tree, so generalized hard-DGM selects the same memory and returns the same prediction.
\end{proof}

\begin{theorem}[DGM data structure, proof sketch for Theorem~\ref{thm:dpm-complexity:formal}]
\label{thm:dpm-complexity:sketch}
The main-text statement is Theorem~\ref{thm:dpm-complexity:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
A prediction computes distances from $h$ to $N_t$ prototypes in $\R^d$, costing
\begin{align*}
O(N_td).
\end{align*}
It then evaluates $Q_k$ selected incident edge gates and $k$ selected local memories, costing $O(Q_k+kd+C_{\mathrm{local}})$ including sparse routing weights. Repair adds the local repair cost. Creating one anchor-centroid pair stores two vectors in $\R^d$, still $O(d)$. Adding $q$ distinction edges stores $q$ vectors and thresholds, costing
\begin{align*}
O(qd).
\end{align*}
For embedding-mean memories, each prototype stores $a_j,c_j\in\R^d$ and $r_j\in\R^{d_e}$, while each edge stores one distinction vector in $\R^d$. Hence memory is
\begin{align*}
O(N_t(2d+d_e)+|E_t|d).
\end{align*}
\end{proof}

\begin{theorem}[DGM layer cost, proof sketch for Theorem~\ref{thm:dgm-layer-cost:formal}]
\label{thm:dgm-layer-cost:sketch}
The main-text statement is Theorem~\ref{thm:dgm-layer-cost:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
Computing $q=Qh$ costs $O(dd_q)$. Retrieval costs $C_{\mathrm{ret}}(N_t,d_q)$ by definition. Edge-gate evaluation over selected incident edges costs $O(Q_k)$. For each selected prototype, a rank-$r$ adapter computes
\begin{align*}
W_jKh.
\end{align*}
The matrix-vector product with one local adapter costs $O(dr)$, so $k$ selected adapters cost
\begin{align*}
O(kdr).
\end{align*}
The detached write path repeats query, retrieval, and edge-gate evaluation, then updates only selected local states. A rank-one local repair or low-rank sufficient-statistic update costs $O(dr)$ per selected memory, with $O(d_q)$ additional prototype bookkeeping. Hence write overhead is
\begin{align*}
O(dd_q+C_{\mathrm{ret}}(N_t,d_q)+Q_k+k(d_q+dr)).
\end{align*}
Because the write path is detached, backpropagation does not need intermediate copies of $\calS_t$ for all $t$. It stores the selected read computation used in the forward loss.
\end{proof}

\begin{theorem}[One-hot memory, proof sketch for Theorem~\ref{thm:one-hot-measurable:formal}]
\label{thm:one-hot-measurable:sketch}
The main-text statement is Theorem~\ref{thm:one-hot-measurable:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
If $\omega,\omega'\in B_i$, then both one-hot features equal $e_i$, so
\begin{align*}
W\phi_\calP(\omega)=We_i=W\phi_\calP(\omega').
\end{align*}
Thus $W\phi_\calP$ is constant on atoms. Conversely, if $f$ is constant on each $B_i$, put that constant value in column $i$ of $W$. Then
\begin{align*}
f(\omega)=W\phi_\calP(\omega)
\end{align*}
for every $\omega$.
\end{proof}

\begin{theorem}[CPM conflict, proof sketch for Theorem~\ref{thm:cpm-conflict-distinction:formal}]
\label{thm:cpm-conflict-distinction:sketch}
The main-text statement is Theorem~\ref{thm:cpm-conflict-distinction:informal}.
\end{theorem}
\begin{proof}[Proof sketch]
If $\omega,\omega'\in B_i$, then
\begin{align*}
\phi_\calP(\omega)=\phi_\calP(\omega')=e_i.
\end{align*}
After committing $We_i=v(\omega)$, the later evidence $We_i=v(\omega')$ has the same key. Its innovation relative to the old key span is
\begin{align*}
z=(I-e_ie_i^\top)e_i=0.
\end{align*}
Its residual is
\begin{align*}
e=v(\omega')-We_i=v(\omega')-v(\omega)\ne0.
\end{align*}
\end{proof}

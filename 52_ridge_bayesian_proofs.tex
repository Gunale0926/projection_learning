\section{DGM Algorithm and Projection-Bridge Proofs}
\label{app:algorithm-bridge-proofs}

This appendix combines elementary finite graph/partition arguments with standard concentration and linear-algebra tools: Hoeffding finite-class validation bounds, nearest-neighbor indexing costs, Moore-Penrose projections, and ridge/Tikhonov least squares \citep{hoeffding1963probability,devroye1996probabilistic,indyk1998approximate,johnson2019billion,penrose1955generalized,tikhonov1977solutions,hoerl1970ridge}.

\begin{theorem}[Anchor edge separation, formal version of Theorem~\ref{thm:anchor-edge:informal}]
\label{thm:anchor-edge:formal}
Let $a_i,a_j\in\R^d$ with $a_i\ne a_j$. Define
\begin{align*}
u=\frac{a_i-a_j}{\norm{a_i-a_j}_2},\qquad b=\frac{1}{2}u^\top(a_i+a_j).
\end{align*}
Then
\begin{align*}
u^\top a_i-b=\frac{1}{2}\norm{a_i-a_j}_2,\qquad u^\top a_j-b=-\frac{1}{2}\norm{a_i-a_j}_2.
\end{align*}
\end{theorem}
\begin{proof}
Compute
\begin{align*}
u^\top a_i-b&=u^\top a_i-\frac{1}{2}u^\top a_i-\frac{1}{2}u^\top a_j\\
&=\frac{1}{2}u^\top(a_i-a_j)\\
&=\frac{1}{2}\norm{a_i-a_j}_2.
\end{align*}
Similarly,
\begin{align*}
u^\top a_j-b&=u^\top a_j-\frac{1}{2}u^\top a_i-\frac{1}{2}u^\top a_j\\
&=-\frac{1}{2}u^\top(a_i-a_j)\\
&=-\frac{1}{2}\norm{a_i-a_j}_2.
\end{align*}
\end{proof}

\begin{theorem}[DGM graph generates a finite information structure, formal version of Theorem~\ref{thm:dgm-graph-partition:informal}]
\label{thm:dgm-graph-partition:formal}
Let $E$ be a finite set of binary generators $g_e:\calH\to\{0,1\}$. Define
\begin{align*}
\sigma_E(h)=\big(g_e(h)\big)_{e\in E}.
\end{align*}
The relation $h\sim_E h'$ iff $\sigma_E(h)=\sigma_E(h')$ is an equivalence relation. Its nonempty equivalence classes form a finite partition $\calP_E^G$ of $\calH$. If a new generator $g$ is added, then
\begin{align*}
\calP_{E\cup\{g\}}^G=\calP_E^G\vee\sigma(g),
\end{align*}
where the right side denotes cutting each atom of $\calP_E^G$ by $g^{-1}(1)$ and $g^{-1}(0)$ and keeping nonempty pieces.
\end{theorem}
\begin{proof}
Equality of signatures is reflexive, symmetric, and transitive, so $\sim_E$ is an equivalence relation. Equivalence classes are disjoint, nonempty, and cover $\calH$, hence they form a partition. Since $E$ is finite, there are at most $2^{|E|}$ signatures, so the partition is finite.

For the update, two points have the same signature under $E\cup\{g\}$ exactly when they have the same old $E$-signature and the same value of $g$. Therefore each new equivalence class is an old atom intersected with either $g^{-1}(1)$ or $g^{-1}(0)$, with empty intersections discarded. This is precisely $\calP_E^G\vee\sigma(g)$.
\end{proof}

\begin{theorem}[Anchor-based hard routing is graph-plus-anchor-order measurable, formal version of Theorem~\ref{thm:hard-dgm-measurable:informal}]
\label{thm:hard-dgm-measurable:formal}
Let $E_t$ be finite and let $a_1,\ldots,a_{N_t}\in\R^d$ be finite frozen anchors with deterministic tie-breaking. Let $\calF_t^{\mathrm{edge}}=\sigma(g_e:e\in E_t)$ and let $\calF_t^{\mathrm{anchor\text{-}order}}$ be generated by all pairwise anchor-distance comparison events
\begin{align*}
\{h:\norm{h-a_i}_2^2\le\norm{h-a_j}_2^2\}.
\end{align*}
The exact hard-route selected concept index, defined as the nearest eligible anchor with fallback to the nearest anchor, is measurable with respect to $\calF_t^{\mathrm{edge}}\vee\calF_t^{\mathrm{anchor\text{-}order}}$.
\end{theorem}
\begin{proof}
For each concept $j$, eligibility is an intersection of finitely many edge-bit events of the form
\begin{align*}
\{h:g_e(h)=1\}
\qquad\text{or}\qquad
\{h:g_e(h)=0\},
\end{align*}
depending on $j$'s recorded side of edge $e$. Hence the eligible set for $j$ is $\calF_t^{\mathrm{edge}}$-measurable.

The anchor-order sigma-algebra determines the complete order of all anchors under the fixed tie-breaking rule. For any subset $A\subseteq\{1,\ldots,N_t\}$, the event that concept $j$ is the nearest anchor in $A$ is a finite intersection of pairwise anchor-comparison events, hence belongs to $\calF_t^{\mathrm{anchor\text{-}order}}$. The selected hard-route index is obtained by combining finitely many eligibility events with these nearest-in-subset events. Therefore the inverse image of each selected index is a finite union of intersections of sets from $\calF_t^{\mathrm{edge}}$ and $\calF_t^{\mathrm{anchor\text{-}order}}$, and belongs to their join.
\end{proof}

\begin{theorem}[Soft routing approaches hard routing, formal version of Theorem~\ref{thm:soft-routing-hard-limit:informal}]
\label{thm:soft-routing-hard-limit:formal}
Let $N$ be a finite candidate set and let
\begin{align*}
\alpha_j=\frac{\exp(-d_j/T)\chi_j}{\sum_{r\in N}\exp(-d_r/T)\chi_r}.
\end{align*}
If $j^\star_{\mathrm{hard}}\notin N$, where $j^\star_{\mathrm{hard}}$ is the exact hard-route selected concept, then this sparse soft read gives no guarantee for the omitted hard-route concept. Otherwise, assume $j^\star=j^\star_{\mathrm{hard}}\in N$.
If $j^\star$ is the unique candidate with $d_j-d_{j^\star}\ge\delta>0$ for all $j\ne j^\star$ and $\chi_j=\chi_{j^\star}=1$, then
\begin{align*}
1-\alpha_{j^\star}\le (|N|-1)\exp(-\delta/T).
\end{align*}
Moreover, suppose $\log\chi_j$ contains a term
\begin{align*}
\frac{\beta_e}{\sqrt{D_j}}
\log\operatorname{sigmoid}\left(\frac{m_j}{\tau_e}\right)
\end{align*}
with $\beta_e>0$ and finite $D_j\ge1$, and candidate $j$ violates that edge with $m_j\le-\delta_e<0$. Then $\chi_j\le \exp(-\beta_e\delta_e/(\sqrt{D_j}\tau_e))$.

For a wrong-side set $A\subseteq N$, suppose every $j\in A$ violates some incident structural edge whose weight is at least $\beta_{\min}>0$, whose signed margin is at most $-\delta_e$, and whose incident degree is at most $D_{\max}$. If at least one eligible reference candidate $r$ has $\chi_r\ge\chi_{\min}>0$, and the distance score ratio between every $j\in A$ and $r$ is bounded by $C$, then the total wrong-side routing mass is at most
\begin{align*}
\frac{C|N|}{\chi_{\min}}
\exp\left(-\frac{\beta_{\min}\delta_e}{\sqrt{D_{\max}}\tau_e}\right).
\end{align*}
\end{theorem}
\begin{proof}
If $j^\star_{\mathrm{hard}}\notin N$, then the soft normalization assigns weight only to candidates in $N$ and the omitted hard-route concept has no route to receive weight. This is the candidate-recall caveat.

For the nearest-reference statement,
\begin{align*}
\frac{\sum_{j\ne j^\star}\exp(-d_j/T)}
{\exp(-d_{j^\star}/T)}
=\sum_{j\ne j^\star}\exp(-(d_j-d_{j^\star})/T)
\le (|N|-1)\exp(-\delta/T).
\end{align*}
Since
\begin{align*}
1-\alpha_{j^\star}
=\frac{\sum_{j\ne j^\star}\exp(-d_j/T)}
{\exp(-d_{j^\star}/T)+\sum_{j\ne j^\star}\exp(-d_j/T)},
\end{align*}
the same upper bound follows.

For an edge violation, set $a=\delta_e/\tau_e>0$. Since
\begin{align*}
\operatorname{sigmoid}(-a)=\frac{1}{1+\exp(a)}\le \exp(-a),
\end{align*}
the corresponding log-compatibility contribution is at most $-\beta_e\delta_e/(\sqrt{D_j}\tau_e)$, and therefore $\chi_j\le \exp(-\beta_e\delta_e/(\sqrt{D_j}\tau_e))$. If $A$ is the wrong-side set and $r$ is an eligible candidate whose compatibility satisfies $\chi_r\ge\chi_{\min}$, the uniform wrong-side assumptions give, for every $j\in A$,
\begin{align*}
\chi_j\le
\exp\left(-\frac{\beta_{\min}\delta_e}{\sqrt{D_{\max}}\tau_e}\right).
\end{align*}
Combining this with the assumed distance-score ratio bound gives
\begin{align*}
\sum_{j\in A}\alpha_j
&\le \sum_{j\in A} \frac{C}{\chi_{\min}}
\exp\left(-\frac{\beta_{\min}\delta_e}{\sqrt{D_{\max}}\tau_e}\right)\\
&\le \frac{C|N|}{\chi_{\min}}
\exp\left(-\frac{\beta_{\min}\delta_e}{\sqrt{D_{\max}}\tau_e}\right).
\end{align*}
\end{proof}

\begin{theorem}[Refine-or-repair objective descent, formal version of Theorem~\ref{thm:refine-repair-descent:informal}]
\label{thm:refine-repair-descent:formal}
Let
\begin{align*}
J_{\calB_t^{\mathrm{score}}}(\calS)
=
\widehat R(\calS;\calB_t^{\mathrm{score}})
+\lambda_{\mathrm{concept}}N_{\mathrm{concept}}(\calS)
+\lambda_{\mathrm{edge}}\abs{E(\calS)}
\end{align*}
with nonnegative penalties on a fixed scoring buffer. If a repair step is committed only after the guarded test
\begin{align*}
\widehat R(\calS_{t+1};\calB_t^{\mathrm{score}})
\le
\widehat R(\calS_t;\calB_t^{\mathrm{score}}),
\end{align*}
and leaves anchors, edges, and concept count unchanged, then $J_{\calB_t^{\mathrm{score}}}$ does not increase. If a refinement creates one concept and $q$ edges and is accepted only when
\begin{align*}
\Delta_{\mathrm{refine}}>
\lambda_{\mathrm{concept}}+\lambda_{\mathrm{edge}}q+\tau+2\epsilon_m,
\end{align*}
where
\begin{align*}
\Delta_{\mathrm{refine}}
=
\widehat R(\calS_t;\calB_t^{\mathrm{score}})
-
\widehat R(\calS_{t+1};\calB_t^{\mathrm{score}}),
\end{align*}
then
\begin{align*}
J_{\calB_t^{\mathrm{score}}}(\calS_{t+1})
<
J_{\calB_t^{\mathrm{score}}}(\calS_t)-\tau-2\epsilon_m.
\end{align*}
\end{theorem}
\begin{proof}
For repair, $N_{\mathrm{concept}}$ and $|E|$ are unchanged. The empirical-risk term on $\calB_t^{\mathrm{score}}$ is no larger by the guard, so $J_{\calB_t^{\mathrm{score}}}$ does not increase. The identity candidate is always feasible, so the guarded rule can decline harmful centroid or memory movement instead of committing it.

For refinement,
\begin{align*}
J_{\calB_t^{\mathrm{score}}}(\calS_t)-J_{\calB_t^{\mathrm{score}}}(\calS_{t+1})
&=\widehat R(\calS_t;\calB_t^{\mathrm{score}})
-\widehat R(\calS_{t+1};\calB_t^{\mathrm{score}})
-\lambda_{\mathrm{concept}}-\lambda_{\mathrm{edge}}q\\
&=\Delta_{\mathrm{refine}}-\lambda_{\mathrm{concept}}-\lambda_{\mathrm{edge}}q\\
&>\tau+2\epsilon_m.
\end{align*}
Rearranging gives the claim.
\end{proof}

The theorem is local to the fixed scoring buffer used in its statement. If the scoring buffer changes over time, these inequalities alone do not define a common global descent potential.

\begin{theorem}[Statistically safe finite-class refinement, formal version of Theorem~\ref{thm:safe-refinement:informal}]
\label{thm:safe-refinement:formal}
Assume losses lie in $[0,1]$. Let $\calH$ be a finite class of candidate refinements fixed before scoring, or generated from a proposal buffer independent of a scoring buffer $\calB^{\mathrm{score}}$ containing $m$ independent samples. For candidate $g$, let $\widehat\Delta(g)$ be the empirical gain on $\calB^{\mathrm{score}}$ and let $\Delta(g)$ be the corresponding population gain. Then, with probability at least $1-\delta$, every $g\in\calH$ satisfies
\begin{align*}
\abs{\widehat\Delta(g)-\Delta(g)}
\le
2\epsilon_m,
\qquad
\epsilon_m=
\sqrt{\frac{\log(2(\abs{\calH}+1)/\delta)}{2m}}.
\end{align*}
Consequently, if
\begin{align*}
\widehat\Delta(g)>
\lambda\operatorname{cost}(g)+\tau+2\epsilon_m,
\end{align*}
then on the same event
\begin{align*}
\Delta(g)>
\lambda\operatorname{cost}(g)+\tau.
\end{align*}
\end{theorem}
\begin{proof}
Apply Hoeffding's inequality to the empirical risk of the old state and to the empirical risk of each candidate-refined state. For any fixed state in this finite collection,
\begin{align*}
\mathbb P\left(\abs{\widehat R-R}>\epsilon\right)
\le 2\exp(-2m\epsilon^2).
\end{align*}
Because $\calH$ is fixed before scoring or generated from data independent of $\calB^{\mathrm{score}}$, Hoeffding's inequality applies conditionally on $\calH$. Taking a union bound over the old state and the $\abs{\calH}$ candidate states gives the displayed $\epsilon_m$. On this event, each gain is a difference of two risks, so empirical and population gain differ by at most $2\epsilon_m$. The implication follows by subtracting this deviation from the empirical gain lower bound.
\end{proof}

\begin{corollary}[Bounded language-model refinement score, formal version of Corollary~\ref{cor:lm-clipped-refinement:informal}]
\label{cor:lm-clipped-refinement:formal}
For language modeling, let
\begin{align*}
\ell_{\mathrm{clip}}(x_{\le t},x_{t+1})
=
\frac{1}{L_{\max}}
\min\{-\log p(x_{t+1}\mid x_{\le t}),L_{\max}\}
\end{align*}
for a fixed $L_{\max}>0$. If the scoring buffer, empirical gain, and population gain in Theorem~\ref{thm:safe-refinement:formal} are computed using $\ell_{\mathrm{clip}}$, then the theorem applies to language-model refinement decisions made with this clipped score. In particular, an accepted refinement has positive clipped population gain after paying complexity cost and margin $\tau$ with probability at least $1-\delta$.
\end{corollary}
\begin{proof}
The clipped score satisfies $0\le \ell_{\mathrm{clip}}\le 1$. Substituting this bounded loss into Theorem~\ref{thm:safe-refinement:formal} gives the claim directly. The statement is only about the clipped score; raw negative log likelihood is unbounded and would require a separate concentration assumption.
\end{proof}

\begin{theorem}[Anchor-stable DGM information structure, formal version of Theorem~\ref{thm:anchor-stable:informal}]
\label{thm:anchor-stable:formal}
Let
\begin{align*}
\calF_t^{\mathrm{DGM}}
=
\sigma(g_e:e\in E_t)
\vee
\sigma\!\left(\{h:\norm{h-a_i}_2^2\le \norm{h-a_j}_2^2\}:1\le i,j\le N_t\right).
\end{align*}
Suppose every edge generator $g_e(h)=\one\{u_e^\top h>b_e\}$ is computed from frozen endpoint anchors $a_i,a_j$. Any repair step that changes only centroids $c_j$, local memories $M_j$, and counts $n_j$, while keeping anchors and edges fixed, leaves $\calF_t^{\mathrm{DGM}}$ unchanged. If a refinement adds a new anchor and new edge generators, then $\calF_{t+1}^{\mathrm{DGM}}$ is $\calF_t^{\mathrm{DGM}}$ joined with the new anchor-order events and the new edge generators.
\end{theorem}
\begin{proof}
For an edge $e=(i,j)$, both
\begin{align*}
u_e=\frac{a_i-a_j}{\norm{a_i-a_j}_2},
\qquad
b_e=\frac{1}{2}u_e^\top(a_i+a_j)
\end{align*}
are deterministic functions of the frozen anchors. A repair step does not change these anchors, so it does not change $u_e$, $b_e$, or $g_e$. It also does not change any anchor-distance comparison event because all anchors are fixed. Hence both sigma-algebras in the join defining $\calF_t^{\mathrm{DGM}}$ are unchanged.

If a refinement adds anchor $a_{\mathrm{new}}$ and edge generators $E_{\mathrm{new}}$, the new edge sigma-algebra is $\sigma(g_e:e\in E_t)\vee\sigma(g_e:e\in E_{\mathrm{new}})$, and the new anchor-order sigma-algebra is the old anchor-order sigma-algebra joined with the pairwise comparisons involving $a_{\mathrm{new}}$. This is exactly the stated join.
\end{proof}

\begin{corollary}[Finite number of accepted refinements, formal version of Corollary~\ref{cor:finite-refinements:informal}]
\label{cor:finite-refinements:formal}
Let $\Phi$ be a common potential with $\Phi(\calS)\ge0$ for all reachable states. If each accepted refinement decreases $\Phi$ by at least $\tau>0$, then the number of accepted refinements is at most $\Phi(\calS_0)/\tau$.
\end{corollary}
\begin{proof}
After $m$ accepted refinements,
\begin{align*}
\Phi(\calS_m)\le \Phi(\calS_0)-m\tau.
\end{align*}
Since $\Phi(\calS_m)\ge0$, we have $m\le \Phi(\calS_0)/\tau$.
\end{proof}

\begin{proposition}[Trees as generalized hard-route DGMs, formal version of Proposition~\ref{prop:tree-as-dgm:informal}]
\label{prop:tree-as-dgm:formal}
Let $\calT$ be a binary decision tree whose internal nodes compute binary distinctions $g_\nu(h)\in\{0,1\}$ and whose leaves have predictors $M_\ell$. Construct a generalized hard-route DGM with one concept $c_\ell$ and memory $M_\ell$ for each leaf $\ell$, where the generalized model permits arbitrary finite binary eligibility constraints in addition to pairwise distinction edges. For each leaf, add hard eligibility constraints equal to the signed distinctions along the root-to-leaf path. If the generalized hard route selects the unique eligible leaf concept, then its prediction equals the tree prediction for every $h$.
\end{proposition}
\begin{proof}
For any input $h$, the tree follows exactly one root-to-leaf path. The leaf reached by the tree is the unique leaf whose path constraints are all satisfied by $h$. By construction, the corresponding generalized hard-route concept has exactly those eligibility constraints, and all other leaf concepts violate at least one path constraint at the first node where their path differs from the tree path. Therefore the generalized hard route selects the same leaf and applies the same memory $M_\ell$.
\end{proof}

\begin{theorem}[DGM data-structure bound, formal version of Theorem~\ref{thm:dgm-complexity:informal}]
\label{thm:dgm-complexity:formal}
Assume DGM has $N_t$ concepts in $\R^d$, exact centroid and anchor retrieval, candidate set $N(h)$, selected incident edge-gate count $Q(h)=\sum_{j\in N(h)}|I_t(j)|$, and local-memory evaluation cost $C_{\mathrm{local}}$. Then one prediction costs $O(N_td+Q(h)+|N(h)|d+C_{\mathrm{local}})$. One observation that repairs a selected concept costs this plus the guarded local repair scoring cost. One observation that creates a new anchor-centroid pair and $q$ distinction edges costs an additional $O(d+qd)$. If each concept stores an anchor, a centroid, and an embedding mean in $\R^{d_e}$, and each edge stores a linear distinction in $\R^d$, then the runtime state uses $O(N_t(2d+d_e)+|E_t|d)$ memory.
\end{theorem}
\begin{proof}
A distance computation between $h\in\R^d$ and one concept center costs $O(d)$. Exact centroid and anchor retrieval over $N_t$ concepts therefore costs
\begin{align*}
O(N_td).
\end{align*}
Graph-neighbor expansion has cost proportional to the incident lists touched and is absorbed into selected edge-gate bookkeeping. After the candidate set $N(h)$ is selected, evaluating the incident edge gates costs $O(Q(h))$. Computing sparse routing weights, distances within the selected set, and evaluating selected local memories costs
\begin{align*}
O(|N(h)|d+C_{\mathrm{local}}).
\end{align*}
Thus prediction costs
\begin{align*}
O(N_td+Q(h)+|N(h)|d+C_{\mathrm{local}}).
\end{align*}
Observation begins with the same retrieval computation. If it repairs a selected concept, it adds the cost of evaluating and scoring the proposed local repair candidates on the scoring buffer. If it creates a new anchor-centroid pair, storing the pair costs $O(d)$. For each new edge, computing and storing the linear distinction vector costs $O(d)$, so $q$ edges cost
\begin{align*}
O(qd).
\end{align*}
The total additional cost for refinement is therefore $O(d+qd)$.

For memory, each concept stores $a_j,c_j\in\R^d$ and $r_j\in\R^{d_e}$, giving
\begin{align*}
O(N_t(2d+d_e)).
\end{align*}
Each edge stores one distinction vector and threshold, giving
\begin{align*}
O(|E_t|d).
\end{align*}
Adding the two terms proves the claimed bound.
\end{proof}

\begin{theorem}[DGM layer cost, formal version of Theorem~\ref{thm:dgm-layer-cost:informal}]
\label{thm:dgm-layer-cost:formal}
Consider a DGM adapter extension with hidden dimension $d$, query dimension $d_q$, local rank $r$, candidate count $|N|$, selected incident edge-gate count $Q_N$, and concept-bank query cost $C_{\mathrm{ret}}(N_t,d_q)$. For one token, the read overhead is
\begin{align*}
O(dd_q+C_{\mathrm{ret}}(N_t,d_q)+Q_N+|N|dr).
\end{align*}
If the write path is detached and uses rank-one local repair, the write overhead is
\begin{align*}
O(dd_q+C_{\mathrm{ret}}(N_t,d_q)+Q_N+|N|(d_q+dr)).
\end{align*}
For a block of length $L$, backpropagation through the slow parameters does not require storing the full DGM state history $\calS_1,\ldots,\calS_L$.
\end{theorem}
\begin{proof}
For a token hidden state $h\in\R^d$, the query computation
\begin{align*}
q=Qh
\end{align*}
costs $O(dd_q)$. The retrieval cost is $C_{\mathrm{ret}}(N_t,d_q)$ by definition. Evaluating selected incident edge gates costs $O(Q_N)$.

For a selected concept $j$, the rank-$r$ adapter read is
\begin{align*}
W_jKh.
\end{align*}
Multiplying one selected $W_j\in\R^{d\times r}$ by a rank-$r$ key costs $O(dr)$. Therefore $|N|$ selected adapters cost
\begin{align*}
O(|N|dr).
\end{align*}
The routing weights add lower-order top-$k$ normalization cost, absorbed into the same bound. This proves the read overhead.

The write path first computes the same query, retrieval, and edge gates. It then updates selected concept statistics and selected local memories. Concept bookkeeping costs $O(d_q)$ per selected concept. A rank-one local repair of a rank-$r$ or low-rank sufficient-statistic memory costs $O(dr)$ per selected memory. Hence the write overhead is
\begin{align*}
O(dd_q+C_{\mathrm{ret}}(N_t,d_q)+Q_N+|N|(d_q+dr)).
\end{align*}

Finally, the update rule is applied as
\begin{align*}
\operatorname{stopgrad}(\calS_{t+1})=\operatorname{DGMObserve}(\operatorname{stopgrad}(\calS_t),h_t,x_{t+1}).
\end{align*}
Thus the computational graph for the loss contains the selected read computation, but not the chain of state-update maps from $\calS_1$ through $\calS_L$. Therefore full DGM state history is not required for backpropagation through the slow parameters.
\end{proof}

\begin{theorem}[One-hot memory is partition-measurable, formal version of Theorem~\ref{thm:one-hot-measurable:informal}]
\label{thm:one-hot-measurable:formal}
Let $\calP=\{B_1,\ldots,B_n\}$ be a finite partition and let $\phi_\calP(\omega)=e_i$ for $\omega\in B_i$. For every $W\in\R^{m\times n}$, the map $f_W(\omega)=W\phi_\calP(\omega)$ is $\calP$-measurable. Conversely, every $\calP$-measurable map $f:\Omega\to\R^m$ is equal to $f_W$ for some $W\in\R^{m\times n}$.
\end{theorem}
\begin{proof}
If $\omega,\omega'\in B_i$, then
\begin{align*}
\phi_\calP(\omega)=\phi_\calP(\omega')=e_i.
\end{align*}
Thus
\begin{align*}
f_W(\omega)=We_i=f_W(\omega'),
\end{align*}
so $f_W$ is constant on each atom and is $\calP$-measurable.

Conversely, if $f$ is $\calP$-measurable, it is constant on each $B_i$. Let $w_i$ be this constant value on $B_i$, and define
\begin{align*}
W=[w_1,\ldots,w_n].
\end{align*}
Then for $\omega\in B_i$,
\begin{align*}
W\phi_\calP(\omega)=We_i=w_i=f(\omega).
\end{align*}
\end{proof}

\begin{theorem}[CPM conflict equals missing distinction, formal version of Theorem~\ref{thm:cpm-conflict-distinction:informal}]
\label{thm:cpm-conflict-distinction:formal}
Let $\calP=\{B_1,\ldots,B_n\}$ and let $\omega,\omega'\in B_i$. Suppose $v(\omega)\ne v(\omega')$. Let $k=\phi_\calP(\omega)=\phi_\calP(\omega')=e_i$. If a memory $W$ has already committed $We_i=v(\omega)$, then the evidence $We_i=v(\omega')$ has innovation $z=0$ relative to the old key span and residual $e\ne0$.
\end{theorem}
\begin{proof}
Since $\omega,\omega'\in B_i$,
\begin{align*}
\phi_\calP(\omega)=\phi_\calP(\omega')=e_i.
\end{align*}
The old key span contains $e_i$. Therefore the innovation of the new key $k=e_i$ relative to the old key matrix $K=e_i$ is
\begin{align*}
z=(I-KK^\dagger)k=(I-e_ie_i^\top)e_i=0.
\end{align*}
The new residual is
\begin{align*}
e=v(\omega')-We_i=v(\omega')-v(\omega).
\end{align*}
Since $v(\omega')\ne v(\omega)$, we have $e\ne0$.
\end{proof}
